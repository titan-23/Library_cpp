116  class Param {
117   private:
118    Timer timer;
119    double time_limit;
120    int beam_depth_base, beam_width_base;
121    int decide_turn, max_turn;
122    vector<int> pred, pred_acc;
123    int done_depth_total;
124    bool adjace;
125
126   public:
127    Param() {}
128
129    /**
130     *
131     @brief Construct a new Param object
132     *
133     * @param time_limit
134     * @param max_turn
135     * @param beam_depth_base
136     * @param beam_width_base
137     */
138    Param(const double time_limit,
139          const int max_turn,
140          const int decide_turn,
141          const int beam_depth_base,
142          const int beam_width_base,
143          const bool adjace=false) :
144        time_limit(time_limit),
145        beam_depth_base(beam_depth_base), beam_width_base(beam_width_base),
146        decide_turn(decide_turn),
147        max_turn(max_turn),
148        pred(max_turn+1, 0), pred_acc(max_turn+2, 0),
149        done_depth_total(0),
150        adjace(adjace) {
151      vector<int> t(max_turn+1);
152      for (int turn = 0; turn < max_turn+1; ++turn) {
153        pred[turn] = max(1, min(beam_depth_base, max_turn-turn));
154        pred_acc[turn+1] = pred_acc[turn] + pred[turn];
155      }
156    }
157
158    void init() { timer.reset(); }
159    int get_max_turn() const { return max_turn; }
160    int get_beam_depth() const { return beam_depth_base; }
161    int get_beam_width() const { return beam_width_base; }
162
163    int get_beam_depth(const int turn) const {
164      return min(beam_depth_base, max_turn-turn);
165    }
166
167    void timestamp(const int turn, const int done_depth) {
168      done_depth_total += done_depth;
169    }
170
171    int get_decide_turn() const {
172      return decide_turn;
173    }
174
175    int get_beam_width(const int turn) {
176      if ((!adjace) || turn == 0) return get_beam_width();
177      double now_time = timer.elapsed();
178      double rem_time = time_limit - now_time;
179      int t = pred_acc.back() - pred_acc[turn];
180      double pred_total_cnt = rem_time * (double)done_depth_total / now_time;
181      int d = max(1, (int)(pred[turn] * pred_total_cnt / (double)t));
182      return d;
183    }

186  class BeamSearch {
187   private:
188    static inline void calc_next_beam(const vector<long long> &keep,
189                                      const int turn,
190                                      __gnu_pbds::gp_hash_table<HashType, uint8_t> &seen,
191                                      vector<long long> &score_keep) {
192      for (const long long now_state: keep) {
193        const vector<Action> &actions = pool.get(now_state)->get_actions(turn);
194        for (const Action &op : actions) {
195          auto [new_score, new_hash] = pool.get(now_state)->try_op(op);
196          if (seen.find(new_hash) != seen.end()) continue;
197          seen[new_hash] = 0;
198          const long long substate = sub_pool.gen();
199          sub_pool.get(substate)->score = new_score;
200          sub_pool.get(substate)->state = now_state;
201          sub_pool.get(substate)->action = op;
202          score_keep.emplace_back(substate);
203        }
204      }
205    }

207   public:
208    static inline Result run_each_turn(Param &param, const bool verbose=false) {
209      Result result;
210
211      const long long best_state = pool.gen();
212      pool.get(best_state)->init();
213      param.init();
214
215      for (int turn = 0; turn < param.get_max_turn(); ++turn) {
216        if (verbose) cout << "# turn : " << turn << endl;
217        const int beam_depth = param.get_beam_depth(turn);
218        const int beam_width = param.get_beam_width(turn);
219        int done_depth = 0;
220        vector<long long> keep = { pool.copy(best_state) };
221        __gnu_pbds::gp_hash_table<HashType, uint8_t> seen;
222        for (int beam_turn = 0; beam_turn < beam_depth; ++beam_turn, ++done_depth) {
223          vector<long long> score_keep;
224          score_keep.reserve(keep.size());
225          calc_next_beam(keep, turn, seen, score_keep);
226          nth_element(score_keep.begin(), score_keep.begin() + min((long long)score_keep.size(), (long long)param.get_beam_width()), score_keep.end(), [&] (const long long &l, const long long &r) {
227            return sub_pool.get(l)->score < sub_pool.get(r)->score;
228          });
229          const Action &op = sub_pool.get(score_keep.front())->action;
230          bool is_all_same = true;
231          vector<long long> new_keep(min((long long)beam_width, (long long)score_keep.size()));
232          for (int i = 0; i < beam_width && i < score_keep.size(); ++i) {
233            const long long state = pool.copy(sub_pool.get(score_keep[i])->state);
234            if (beam_turn == 0) {
235              pool.get(state)->first_action = sub_pool.get(score_keep[i])->action;
236            }
237            pool.get(state)->apply_op(sub_pool.get(score_keep[i])->action);
238            if (is_all_same && pool.get(state)->first_action != op) is_all_same = false;
239            new_keep[i] = state;
240          }
241          for (const long long state: score_keep) sub_pool.del(state);
242          for (const long long state: keep) pool.del(state);
243          swap(keep, new_keep);
244          const long long d_best_state = *min_element(keep.begin(), keep.end(), [&] (const long long &l, const long long &r) {
245            return (*pool.get(l)) < (*pool.get(r));
246          });
247          if (pool.get(d_best_state)->is_done() || is_all_same) break;
248        }
249        param.timestamp(turn, done_depth);
250        const long long d_best_state = *min_element(keep.begin(), keep.end(), [&] (const long long &l, const long long &r) {
251          return (*pool.get(l)) < (*pool.get(r));
252        });
253        const Action &op = pool.get(d_best_state)->first_action;
254        pool.get(best_state)->apply_op(op);
255        result.history.push_back(op);
256        if (verbose) {
257          pool.get(best_state)->print();
258          cout << "Score = " << pool.get(best_state)->get_score() << endl << endl;
259        }
260        for (const long long node_id: keep) pool.del(node_id);
261        if (pool.get(best_state)->is_done()) break;
262      }
263      result.score = pool.get(best_state)->get_score();
264      pool.del(best_state);
265      return result;
266    }

268    static inline Result run_normal(Param &param, const bool verbose=false) {
269      param.init();
270      Result result;
271
272      __gnu_pbds::gp_hash_table<HashType, uint8_t> seen;
273      vector<long long> keep;
274
275      { // init state
276        const long long state = pool.gen();
277        pool.get(state)->init();
278        seen[state] = 0;
279        keep = { state };
280        pool.get(state)->substate_id = -1;
281      }
282
283      for (int turn = 0; turn < param.get_max_turn(); ++turn) {
284        if (verbose) cout << "# turn : " << turn << endl;
285        const int beam_width = param.get_beam_width(turn);
286        vector<long long> score_keep;
287        calc_next_beam(keep, turn, seen, score_keep);
288        nth_element(score_keep.begin(), score_keep.begin() + min((long long)score_keep.size(), (long long)param.get_beam_width()), score_keep.end(), [&] (const long long &l, const long long &r) {
289          return sub_pool.get(l)->score < sub_pool.get(r)->score;
290        });
291        vector<long long> new_keep(min((long long)beam_width, (long long)score_keep.size()));
292        for (int i = 0; i < beam_width && i < score_keep.size(); ++i) {
293          long long substate = score_keep[i];
294          long long new_state = pool.copy(sub_pool.get(substate)->state);
295          pool.get(new_state)->apply_op(sub_pool.get(substate)->action);
296          pool.get(new_state)->substate_id = substate;
297          sub_pool.get(substate)->par = pool.get(sub_pool.get(substate)->state)->substate_id;
298          new_keep[i] = new_state;
299        }
300
301        for (const long long state: keep) pool.del(state);
302        for (int i = new_keep.size(); i < score_keep.size(); ++i) {
303          sub_pool.del(score_keep[i]);
304        }
305
306        swap(keep, new_keep);
307        const long long best_state = *min_element(keep.begin(), keep.end(), [&] (const long long &l, const long long &r) {
308          return (*pool.get(l)) < (*pool.get(r));
309        });
310        if (verbose) {
311          // pool.get(best_state)->print();
312          cout << "Score = " << pool.get(best_state)->get_score() << endl << endl;
313        }
314        if (pool.get(best_state)->is_done()) break;
315      }
316
317      const long long best_state = *min_element(keep.begin(), keep.end(), [&] (const long long &l, const long long &r) {
318        return (*pool.get(l)) < (*pool.get(r));
319      });
320      result.score = pool.get(best_state)->get_score();
321      long long substate = pool.get(best_state)->substate_id;
322      while (substate != -1) {
323        result.history.emplace_back(sub_pool.get(substate)->action);
324        substate = sub_pool.get(substate)->par;
325      }
326      reverse(result.history.begin(), result.history.end());
327      return result;
328    }

330    static inline Result run_complex(Param &param, const bool verbose=false) {
331      /*
332      - 一度のビームサーチで、何ターン先を決めるか
333        - 幅、深さ
334        - 何ターン先まで読むか
335      */
336
337      Result result;
338      param.init();
339
340      const long long best_state = pool.gen();
341      pool.get(best_state)->init();
342      pool.get(best_state)->substate_id = -1;
343
344      for (int turn = 0; turn < param.get_max_turn(); turn += param.get_decide_turn()) {
345        if (verbose) cout << "# turn : " << turn << endl;
346
347        vector<long long> keep;
348        const int beam_depth = param.get_beam_depth(turn);
349        const int beam_width = param.get_beam_width(turn);
350        __gnu_pbds::gp_hash_table<HashType, uint8_t> seen;
351
352        { // init
353          const long long init_state = pool.copy(best_state);
354          seen[init_state] = 0;
355          pool.get(init_state)->substate_id = -1;
356          keep = {init_state};
357        }
358
359        for (int beam_turn = 0; beam_turn < beam_depth; ++beam_turn) {
360          vector<long long> score_keep;
361          calc_next_beam(keep, turn, seen, score_keep);
362          const int w = min((int)score_keep.size(), beam_width);
363          nth_element(score_keep.begin(), score_keep.begin() + w, score_keep.end(), [&] (const long long &l, const long long &r) {
364            return sub_pool.get(l)->score < sub_pool.get(r)->score;
365          });
366          vector<long long> new_keep;
367
368          for (int i = 0; i < w; ++i) {
369            long long substate = score_keep[i];
370            long long nowstate = sub_pool.get(substate)->state;
371            long long new_state = pool.copy(nowstate);
372            pool.get(new_state)->apply_op(sub_pool.get(substate)->action);
373            pool.get(new_state)->substate_id = substate;
374            sub_pool.get(substate)->par = pool.get(nowstate)->substate_id;
375            new_keep.emplace_back(new_state);
376          }
377
378          for (const long long state: keep) pool.del(state);
379          for (int i = w; i < score_keep.size(); ++i) {
380            sub_pool.del(score_keep[i]);
381          }
382
383          swap(keep, new_keep);
384          const long long this_best_state = *min_element(keep.begin(), keep.end(), [&] (const long long &l, const long long &r) {
385            return (*pool.get(l)) < (*pool.get(r));
386          });
387          if (pool.get(this_best_state)->is_done()) break;
388        }
389
390        const long long this_best_state = *min_element(keep.begin(), keep.end(), [&] (const long long &l, const long long &r) {
391          return (*pool.get(l)) < (*pool.get(r));
392        });
393        long long substate = pool.get(this_best_state)->substate_id;
394        vector<Action> history;
395        while (substate != -1) {
396          history.emplace_back(sub_pool.get(substate)->action);
397          substate = sub_pool.get(substate)->par;
398        }
399        sub_pool.clear();
400        reverse(history.begin(), history.end());
401        for (int i = 0; i < min(param.get_decide_turn(), (int)(history.size())); ++i) {
402          result.history.emplace_back(history[i]);
403          pool.get(best_state)->apply_op(history[i]);
404          if (verbose) {
405            pool.get(best_state)->print();
406            cout << "Score = " << pool.get(best_state)->get_score() << endl << endl;
407          }
408        }
409        if (pool.get(best_state)->is_done()) break;
410      }
411      result.score = pool.get(best_state)->get_score();
412      return result;
413    }

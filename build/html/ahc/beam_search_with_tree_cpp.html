 85//! ビームサーチの過程を表す木のノード
 86struct TreeNode {
 87    TreeNodeID par;
 88    Action pre_action;
 89    bool is_valid;
 90    ScoreType score;
 91    vector<TreeNodeID> child;
 92
 93    TreeNode() : par(-1), is_valid(true) {}
 94
 95    bool is_leaf() const {
 96        return child.empty();
 97    }
 98};

100titan23::StatePool<TreeNode> treenode_pool;
101titan23::StatePool<SubState> substate_pool;

104class BeamSearchWithTree {
105  private:
106    ScoreType best_score;
107    TreeNodeID best_id;
108    titan23::HashSet seen;
109
110    void get_next_beam_recursion(State* state, TreeNodeID node, vector<SubStateID> &next_beam, int depth, const int beam_width) {
111        if (!treenode_pool.get(node)->is_valid) assert(false);
112        if (depth == 0) { // 葉
113            assert(treenode_pool.get(node)->is_leaf());
114            vector<Action> actions = state->get_actions();
115            for (Action &action : actions) {
116                auto [score, hash] = state->try_op(action);
117                SubStateID substate = substate_pool.gen();
118                substate_pool.get(substate)->par = node;
119                substate_pool.get(substate)->action = action;
120                substate_pool.get(substate)->score = score;
121                substate_pool.get(substate)->hash = hash;
122                next_beam.emplace_back(substate);
123            }
124            return;
125        }
126        for (TreeNodeID nxt_node : treenode_pool.get(node)->child) {
127            if (!treenode_pool.get(nxt_node)->is_valid) continue;
128            state->apply_op(treenode_pool.get(nxt_node)->pre_action);
129            get_next_beam_recursion(state, nxt_node, next_beam, depth-1, beam_width);
130            state->rollback(treenode_pool.get(nxt_node)->pre_action);
131        }
132    }
133
134    tuple<int, TreeNodeID, vector<SubStateID>> get_next_beam(State* state, TreeNodeID node, int turn, const int beam_width) {
135        int cnt = 0;
136        while (true) { // 一本道は行くだけ
137            if (treenode_pool.get(node)->child.size() != 1) break;
138            ++cnt;
139            node = treenode_pool.get(node)->child[0];
140            state->apply_op(treenode_pool.get(node)->pre_action);
141        }
142        vector<SubStateID> next_beam;
143        seen.clear();
144        get_next_beam_recursion(state, node, next_beam, turn-cnt, beam_width);
145        return make_tuple(cnt, node, next_beam);
146    }
147
148    //! 親を返す / 無ければ自分を返す(!?)
149    TreeNodeID get_par(SubStateID s_node, int cnt=1) {
150        assert(0 <= s_node && s_node < substate_pool.get_size());
151        TreeNodeID node = substate_pool.get(s_node)->par;
152        assert(node != -1);
153        for (int i = 0; i < cnt; ++i) {
154            assert(node < treenode_pool.get_size());
155            if (treenode_pool.get(node)->par == -1) {
156                assert(node != -1);
157                return node;
158            }
159            assert(node != -1);
160            node = treenode_pool.get(node)->par;
161        }
162        assert(node != -1);
163        assert(node < treenode_pool.get_size());
164        return node;
165    }
166
167    //! 木の各ノードのis_validの更新と、必要なければdelする
168    bool update_tree(const TreeNodeID node, int depth) {
169        if (!treenode_pool.get(node)->is_valid) return false;
170        if (treenode_pool.get(node)->is_leaf()) {
171            if (depth == 0) return true;
172            treenode_pool.get(node)->is_valid = false;
173            return false;
174        }
175        int idx = 0;
176        while (idx < treenode_pool.get(node)->child.size()) {
177            TreeNodeID nxt_node = treenode_pool.get(node)->child[idx];
178            if (!update_tree(nxt_node, depth-1)) {
179                treenode_pool.del(treenode_pool.get(node)->child[idx]);
180                treenode_pool.get(node)->child.erase(treenode_pool.get(node)->child.begin() + idx);
181            } else {
182                ++idx;
183            }
184        }
185        if (idx == 0) {
186            assert(treenode_pool.get(node)->child.empty());
187            treenode_pool.get(node)->is_valid = false;
188            return false;
189        }
190        return true;
191    }
192
193    //! node以上のパスを返す
194    vector<Action> get_path(TreeNodeID node) {
195        vector<Action> result;
196        while (node != -1 && treenode_pool.get(node)->par != -1) {
197            result.emplace_back(treenode_pool.get(node)->pre_action);
198            node = treenode_pool.get(node)->par;
199        }
200        reverse(result.begin(), result.end());
201        return result;
202    }
203
204    //! for debug
205    void print_tree(State* state, const TreeNodeID node, int depth) {
206    }
207
208    //! node以下で、葉かつ最も評価値の良いノードを見るける / 葉はターン数からは判断していないので注意
209    void get_best_node(TreeNodeID node) {
210        if (!treenode_pool.get(node)->is_valid) return;
211        if (treenode_pool.get(node)->is_leaf()) {
212            if (best_id == -1 || treenode_pool.get(node)->score < best_score) {
213                best_score = treenode_pool.get(node)->score;
214                best_id = node;
215            }
216            return;
217        }
218        for (TreeNodeID nxt_node : treenode_pool.get(node)->child) {
219            get_best_node(nxt_node);
220        }
221    }
222
223    vector<Action> get_result(TreeNodeID root) {
224        best_id = -1; // 更新
225        get_best_node(root);
226        TreeNodeID node = best_id;
227        vector<Action> result = get_path(node);
228        cerr << treenode_pool.get_size() << endl;
229        return result;
230    }

232  public:
233    /**
234     * @brief ビームサーチをする
235     *
236     * @param param ターン数、ビーム幅を指定するパラメータ構造体
237     * @param verbose 途中結果のスコアを標準エラー出力するかどうか
238     * @return vector<Action>
239     */
240    vector<Action> search(const Param &param, const bool verbose = false) {
241        TreeNodeID root = treenode_pool.gen();
242        treenode_pool.get(root)->child.clear();
243        treenode_pool.get(root)->par = -1;
244        treenode_pool.get(root)->is_valid = true;
245
246        State* state = new State;
247        state->init();
248
249        this->seen = titan23::HashSet(param.BEAM_WIDTH * 4); // TODO
250
251        int now_turn = 0;
252
253        for (int turn = 0; turn < param.MAX_TURN; ++turn) {
254            if (verbose) cerr << "# turn : " << turn << endl;
255
256            // 次のビーム候補を求める
257            auto [apply_only_turn, next_root, next_beam] = get_next_beam(state, root, turn-now_turn, param.BEAM_WIDTH);
258            root = next_root;
259            now_turn += apply_only_turn;
260            assert(!next_beam.empty());
261            if (verbose) {
262                cerr << "min_score=" << substate_pool.get((*min_element(next_beam.begin(), next_beam.end(), [] (const SubStateID &left, const SubStateID &right) {
263                    return substate_pool.get(left)->score < substate_pool.get(right)->score;
264                })))->score << endl;
265            }
266
267            // ビームを絞る // TODO 評価値が一致した場合、親の評価値も参考にするなど
268            int beam_width = min(param.BEAM_WIDTH, (int)next_beam.size());
269            nth_element(next_beam.begin(), next_beam.begin() + beam_width, next_beam.end(), [&] (const SubStateID &left, const SubStateID &right) {
270                // if (substate_pool.get(left)->score != substate_pool.get(right)->score) {
271                    return substate_pool.get(left)->score < substate_pool.get(right)->score;
272                // }
273                // return treenode_pool.get(get_par(left))->score < treenode_pool.get(get_par(right))->score;
274            });
275
276            // 探索木の更新
277            for (int i = 0; i < beam_width; ++i) {
278                SubStateID s = next_beam[i];
279                TreeNodeID new_node = treenode_pool.gen();
280                treenode_pool.get(substate_pool.get(s)->par)->child.push_back(new_node);
281                treenode_pool.get(new_node)->par = substate_pool.get(s)->par;
282                treenode_pool.get(new_node)->pre_action = substate_pool.get(s)->action;
283                treenode_pool.get(new_node)->score = substate_pool.get(s)->score;
284                treenode_pool.get(new_node)->is_valid = true;
285            }
286            substate_pool.clear();
287            update_tree(root, turn-now_turn+1);
288        }
289
290        // 答えを復元する
291        vector<Action> result = get_result(root);
292        return result;
293    }

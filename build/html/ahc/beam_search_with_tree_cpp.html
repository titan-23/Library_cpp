112     * @brief (疑似)乱数生成クラス(XOR shift)

114    class Random {
115      private:
116        unsigned int _x, _y, _z, _w;
117
118        unsigned int _xor128() {
119            const unsigned int t = _x ^ (_x << 11);
120            _x = _y;
121            _y = _z;
122            _z = _w;
123            _w = (_w ^ (_w >> 19)) ^ (t ^ (t >> 8));
124            return _w;
125        }
126
127      public:
128        Random() : _x(123456789),
129                   _y(362436069),
130                   _z(521288629),
131                   _w(88675123) {}
132
133        //! `[0, 1]` の乱数を返す(実数)
134        double random() { return (double)(_xor128()) / 0xFFFFFFFF; }
135
136        //! `[0, end]` の乱数を返す(整数)
137        int randint(const int end) {
138            assert(0 <= end);
139            return (((unsigned long long)_xor128() * (end+1)) >> 32);
140        }
141
142        //! `[begin, end]` の乱数を返す(整数)
143        int randint(const int begin, const int end) {
144            assert(begin <= end);
145            return begin + (((unsigned long long)_xor128() * (end-begin+1)) >> 32);
146        }
147
148        //! `[0, end)` の乱数を返す(整数)
149        int randrange(const int end) {
150            assert(0 < end);
151            return (((unsigned long long)_xor128() * end) >> 32);
152        }
153
154        //! `[begin, end)` の乱数を返す(整数)
155        int randrange(const int begin, const int end) {
156            assert(begin < end);
157            return begin + (((unsigned long long)_xor128() * (end-begin)) >> 32);
158        }
159
160        //! `[0, u64_MAX)` の乱数を返す / zobrist hash等の使用を想定
161        unsigned long long rand_u64() {
162            return ((unsigned long long)_xor128() << 32) | _xor128();
163        }
164
165        //! `[0, end)` の異なる乱数を2つ返す
166        pair<int, int> rand_pair(const int end) {
167            assert(end >= 2);
168            int u = randrange(end);
169            int v = u + randrange(1, end);
170            if (v >= end) v -= end;
171            if (u > v) swap(u, v);
172            return {u, v};
173        }
174
175        //! `[begin, end)` の異なる乱数を2つ返す
176        pair<int, int> rand_pair(const int begin, const int end) {
177            assert(end - begin >= 2);
178            int u = randrange(begin, end);
179            int v = (u + randrange(1, end-begin));
180            if (v >= end) v -= (end-begin);
181            if (u > v) swap(u, v);
182            return {u, v};
183        }
184
185        //! Note `a`は非const
186        vector<int> rand_vec(const int cnt, vector<int> &a) {
187            int n = a.size();
188            for (int i = 0; i < cnt; ++i) {
189                int j = randrange(i, n);
190                swap(a[i], a[j]);
191            }
192            vector<int> res(a.begin(), a.begin()+cnt);
193            return res;
194        }
195
196        //! `[begin, end)` の乱数を返す(実数)
197        double randdouble(const double begin, const double end) {
198            assert(begin < end);
199            return begin + random() * (end-begin);
200        }
201
202        //! `vector` をインプレースにシャッフルする / `O(n)`
203        template <typename T>
204        void shuffle(vector<T> &a) {
205            int n = a.size();
206            for (int i = 0; i < n-1; ++i) {
207                int j = randrange(i, n);
208                swap(a[i], a[j]);
209            }
210        }
211
212        template <typename T>
213        vector<T> choices(const vector<T> &a, const int k) {
214            assert(a.size() >= k);
215            vector<T> r(k);
216            unordered_set<int> seen;
217            for (int i = 0; i < k; ++i) {
218                int x = randrange(a.size());
219                while (seen.find(x) != seen.end()) x = randrange(a.size());
220                seen.insert(x);
221                r[i] = a[x];
222            }
223            return r;
224        }
225
226        template <typename T>
227        T choice(const vector<T> &a) {
228            return a[randrange(a.size())];
229        }
230
231        template <typename T>
232        T choice(const vector<T> &a, const vector<int> &w, bool normal) {
233            assert(normal == false);
234            assert(a.size() == w.size());
235            double sum = 0.0;
236            for (const int &x: w) sum += x;
237            assert(sum > 0);
238            vector<double> x(w.size());
239            for (int i = 0; i < x.size(); ++i) {
240                x[i] = (double)w[i] / sum;
241                if (i-1 >= 0) x[i] += x[i-1];
242            }
243            return choice(a, x);
244        }
245
246        template <typename T>
247        T choice(const vector<T> &a, const vector<double> &w) {
248            double i = random();
249            int l = -1, r = a.size()-1;
250            while (r - l > 1) {
251                int mid = (l + r) / 2;
252                if (w[mid] <= i) l = mid;
253                else r = mid;
254            }
255            return a[r];
256        }

271    template<typename T>
272    class StatePool {
273      private:
274        vector<T*> pool;
275        stack<int> unused_indx;
276
277      public:
278        StatePool() {}
279        StatePool(const int n) { init(n); }
280
281        //! clear
282        void clear() {
283            while (!unused_indx.empty()) unused_indx.pop();
284            for (int i = (int)pool.size()-1; i >= 0; --i) {
285                unused_indx.emplace(i);
286            }
287        }
288
289        //! n要素確保する。
290        void init(const int n) {
291            for (int i = 0; i < n; ++i) {
292                T* state = new T;
293                pool.emplace_back(state);
294                unused_indx.emplace(i);
295            }
296        }
297
298        //! id に対応する T のポインタを返す。
299        T* get(int id) const {
300            assert(0 <= id && id < pool.size());
301            return pool[id];
302        }
303
304        //! idに対応するTを仮想的に削除する。
305        void del(int id) {
306            assert(0 <= id && id < pool.size());
307            unused_indx.emplace(id);
308        }
309
310        //! T を作成し、それに対応する id を返す。
311        int gen() {
312            int state_id;
313            if (unused_indx.empty()) {
314                T* state = new T;
315                state_id = pool.size();
316                pool.emplace_back(state);
317            } else {
318                state_id = unused_indx.top();
319                unused_indx.pop();
320            }
321            return state_id;
322        }
323
324        //! id に対応するTをコピーし、コピー先のidを返す。
325        //! T のコピーメソッドを呼び出す。
326        int copy(const int id) {
327            int new_id = gen();
328            pool[id]->copy(pool[new_id]);
329            return new_id;
330        }
331
332        //! 内部サイズを呼び出す
333        int get_size() const {
334            return pool.size();
335        }
336    };
337}  // namespace titan23

385    class HashSet {
386      private:
387        using u64 = unsigned long long;
388        static constexpr const u64 K = 0x517cc1b727220a95;
389        static constexpr const int M = 2;
390        vector<u64> exist;
391        vector<u64> keys;
392        int msk, xor_;
393        int size;
394
395        int hash(const u64 &key) const {
396            return (((((key>>32)&msk) ^ (key&msk) ^ xor_)) * (HashSet::K & msk)) & msk;
397        }
398
399        pair<int, bool> get_pos(const u64 &key) const {
400            int h = hash(key);
401            while (true) {
402                if (!(exist[h>>6]>>(h&63)&1)) return {h, false};
403                if (keys[h] == key) return {h, true};
404                h = (h + 1) & msk;
405            }
406        }
407
408        int bit_length(const int x) const {
409            if (x == 0) return 0;
410            return 32 - __builtin_clz(x);
411        }
412
413        void rebuild() {
414            vector<u64> old_exist = exist;
415            vector<u64> old_keys = keys;
416            exist.resize(HashSet::M*old_exist.size()+1);
417            fill(exist.begin(), exist.end(), 0);
418            keys.resize(HashSet::M*old_keys.size());
419            size = 0;
420            msk = (1<<bit_length(keys.size()-1))-1;
421            random_device rd;
422            mt19937 gen(rd());
423            uniform_int_distribution<int> dis(0, msk);
424            xor_ = dis(gen);
425            for (int i = 0; i < (int)old_keys.size(); ++i) {
426                if (old_exist[i>>6]>>(i&63)&1) {
427                    insert(old_keys[i]);
428                }
429            }
430        }
431
432      public:
433        HashSet() : exist(1, 0), keys(1), msk(0), xor_(0), size(0) {}
434
435        HashSet(const int n) {
436            int s = 1<<bit_length(n);
437            s *= HashSet::M;
438            assert(s > 0);
439            exist.resize((s>>6)+1, 0);
440            keys.resize(s);
441            msk = (1<<bit_length(keys.size()-1))-1;
442            random_device rd;
443            mt19937 gen(rd());
444            uniform_int_distribution<int> dis(0, msk);
445            xor_ = dis(gen);
446            size = 0;
447        }
448
449        bool contains(const u64 key) const {
450            return get_pos(key).second;
451        }
452
453        void insert(const u64 key) {
454            const auto [pos, is_exist] = get_pos(key);
455            if (!is_exist) {
456                exist[pos>>6] |= 1ull<<(pos&63);
457                keys[pos] = key;
458                ++size;
459                if (HashSet::M*size > keys.size()) {
460                    rebuild();
461                }
462            }
463        }
464
465        //! keyがすでにあればtrue, なければ挿入してfalse / `O(1)`
466        bool contains_insert(const u64 key) {
467            const auto [pos, is_exist] = get_pos(key);
468            if (!is_exist) {
469                exist[pos>>6] |= 1ull<<(pos&63);
470                keys[pos] = key;
471                ++size;
472                if (HashSet::M*size > keys.size()) {
473                    rebuild();
474                }
475                return false;
476            }
477            return true;
478        }
479
480        //! 全ての要素を削除する / `O(n/w)`
481        void clear() {
482            this->size = 0;
483            fill(exist.begin(), exist.end(), 0);
484        }
485
486        int len() const {
487            return size;
488        }
489    };

586  private:

591    void get_next_beam_recursion(State* state, TreeNodeID node, vector<SubStateID> &next_beam, int depth, const int beam_width) {
592        if (depth == 0) { // 葉
593            vector<Action> actions = state->get_actions();
594            for (Action &action : actions) {
595                auto [score, hash] = state->try_op(action);
596                if (seen.contains_insert(hash)) continue;
597                SubStateID substate = substate_pool.gen();
598                substate_pool.get(substate)->par = node;
599                substate_pool.get(substate)->action = action;
600                substate_pool.get(substate)->score = score;
601                next_beam.emplace_back(substate);
602            }
603            return;
604        }
605        for (TreeNodeID nxt_node : treenode_pool.get(node)->child) {
606            assert(treenode_pool.get(nxt_node)->is_valid);
607        for (const TreeNodeID nxt_node : treenode_pool.get(node)->child) {
608            state->apply_op(treenode_pool.get(nxt_node)->pre_action);
609            get_next_beam_recursion(state, nxt_node, next_beam, depth-1, beam_width);
610            state->rollback(treenode_pool.get(nxt_node)->pre_action);
611        }
612    }
613
614    tuple<int, TreeNodeID, vector<SubStateID>> get_next_beam(State* state, TreeNodeID node, int turn, const int beam_width) {
615        int cnt = 0;
616        while (true) { // 一本道は行くだけ
617            if (treenode_pool.get(node)->child.size() != 1) break;
618            ++cnt;
619            node = treenode_pool.get(node)->child[0];
620            state->apply_op(treenode_pool.get(node)->pre_action);
621        }
622        vector<SubStateID> next_beam;
623        seen.clear();
624        get_next_beam_recursion(state, node, next_beam, turn-cnt, beam_width);
625        return make_tuple(cnt, node, next_beam);
626    }

628    //! 親を返す / 無ければ自分を返す(!?)
629    TreeNodeID get_par(SubStateID s_node, int cnt=1) {
630        assert(0 <= s_node && s_node < substate_pool.get_size());
631        TreeNodeID node = substate_pool.get(s_node)->par;
632        assert(node != -1);
633        for (int i = 0; i < cnt; ++i) {
634            assert(node < treenode_pool.get_size());
635            if (treenode_pool.get(node)->par == -1) {
636                assert(node != -1);
637                return node;
638            }
639            assert(node != -1);
640            node = treenode_pool.get(node)->par;
641        }
642        assert(node != -1);
643        assert(node < treenode_pool.get_size());
644        return node;
645    }

647    //! 不要なNodeを削除し、木を更新する
648    bool update_tree(const TreeNodeID node, const int depth) {
649        if (treenode_pool.get(node)->is_leaf()) return depth == 0;
650        int idx = 0;
651        while (idx < treenode_pool.get(node)->child.size()) {
652            TreeNodeID nxt_node = treenode_pool.get(node)->child[idx];
653            if (!update_tree(nxt_node, depth-1)) {
654                treenode_pool.del(nxt_node);
655                treenode_pool.get(node)->child.erase(treenode_pool.get(node)->child.begin() + idx);
656                continue;
657            }
658            ++idx;
659        }
660        return idx > 0;
661    }

663    //! node以上のパスを返す
664    vector<Action> get_path(TreeNodeID node) {
665        vector<Action> result;
666        while (node != -1 && treenode_pool.get(node)->par != -1) {
667            result.emplace_back(treenode_pool.get(node)->pre_action);
668            node = treenode_pool.get(node)->par;
669        }
670        reverse(result.begin(), result.end());
671        return result;
672    }

678    //! node以下で、葉かつ最も評価値の良いノードを見るける / 葉はターン数からは判断していないので注意
679    void get_best_node(TreeNodeID node) {
680        if (treenode_pool.get(node)->is_leaf()) {
681            if (best_id == -1 || treenode_pool.get(node)->score < best_score) {
682                best_score = treenode_pool.get(node)->score;
683                best_id = node;
684            }
685            return;
686        }
687        for (TreeNodeID nxt_node : treenode_pool.get(node)->child) {
688            get_best_node(nxt_node);
689        }
690    }

692    vector<Action> get_result(TreeNodeID root) {
693        best_id = -1; // 更新
694        get_best_node(root);
695        TreeNodeID node = best_id;
696        vector<Action> result = get_path(node);
697        cerr << treenode_pool.get_size() << endl;
698        return result;

702    /**
703     * @brief ビームサーチをする
704     *
705     * @param param ターン数、ビーム幅を指定するパラメータ構造体
706     * @param verbose 途中結果のスコアを標準エラー出力するかどうか
707     * @return vector<Action>
708     */
709    vector<Action> search(const BeamParam &param, const bool verbose = false) {
710        TreeNodeID root = treenode_pool.gen();
711        treenode_pool.get(root)->child.clear();
712        treenode_pool.get(root)->par = -1;
713
714        State* state = new State;
715        state->init();
716
717        this->seen = titan23::HashSet(param.BEAM_WIDTH * 4); // TODO
718
719        int now_turn = 0;
720
721        for (int turn = 0; turn < param.MAX_TURN; ++turn) {
722            if (verbose) cerr << "# turn : " << turn << endl;
723
724            // 次のビーム候補を求める
725            auto [apply_only_turn, next_root, next_beam] = get_next_beam(state, root, turn-now_turn, param.BEAM_WIDTH);
726            root = next_root;
727            now_turn += apply_only_turn;
728            assert(!next_beam.empty());
729            if (verbose) {
730                cerr << "min_score=" << substate_pool.get((*min_element(next_beam.begin(), next_beam.end(), [] (const SubStateID &left, const SubStateID &right) {
731                    return substate_pool.get(left)->score < substate_pool.get(right)->score;
732                })))->score << endl;
733            }
734
735            // ビームを絞る // TODO 評価値が一致した場合、親の評価値も参考にするなど
736            int beam_width = min(param.BEAM_WIDTH, (int)next_beam.size());
737            nth_element(next_beam.begin(), next_beam.begin() + beam_width, next_beam.end(), [&] (const SubStateID &left, const SubStateID &right) {
738                // if (substate_pool.get(left)->score != substate_pool.get(right)->score) {
739                    return substate_pool.get(left)->score < substate_pool.get(right)->score;
740                // }
741                // return treenode_pool.get(get_par(left))->score < treenode_pool.get(get_par(right))->score;
742            });
743
744            // 探索木の更新
745            for (int i = 0; i < beam_width; ++i) {
746                SubStateID s = next_beam[i];
747                TreeNodeID new_node = treenode_pool.gen();
748                treenode_pool.get(substate_pool.get(s)->par)->child.emplace_back(new_node);
749                treenode_pool.get(new_node)->par = substate_pool.get(s)->par;
750                treenode_pool.get(new_node)->pre_action = substate_pool.get(s)->action;
751                treenode_pool.get(new_node)->score = substate_pool.get(s)->score;
752            }
753            substate_pool.clear();
754            update_tree(root, turn-now_turn+1);
755        }
756
757        // 答えを復元する
758        vector<Action> result = get_result(root);
759        return result;
760    }

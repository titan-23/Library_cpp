 17    class Timer {
 18      private:
 19        chrono::time_point<chrono::high_resolution_clock> start_timepoint;
 20
 21      public:
 22        Timer() : start_timepoint(chrono::high_resolution_clock::now()) {}
 23
 24        //! リセットする
 25        void reset() {
 26            start_timepoint = chrono::high_resolution_clock::now();
 27        }
 28
 29        //! 経過時間[ms](double)を返す
 30        double elapsed() const {
 31            auto end_timepoint = chrono::high_resolution_clock::now();
 32            auto start = chrono::time_point_cast<chrono::microseconds>(start_timepoint).time_since_epoch().count();
 33            auto end = chrono::time_point_cast<chrono::microseconds>(end_timepoint).time_since_epoch().count();
 34            return (end - start) * 0.001;
 35        }
 36    };

 48    /**
 49     * @brief (疑似)乱数生成クラス(XOR shift)
 50     */
 51    class Random {
 52      private:
 53        unsigned int _x, _y, _z, _w;
 54
 55        unsigned int _xor128() {
 56            const unsigned int t = _x ^ (_x << 11);
 57            _x = _y;
 58            _y = _z;
 59            _z = _w;
 60            _w = (_w ^ (_w >> 19)) ^ (t ^ (t >> 8));
 61            return _w;
 62        }
 63
 64      public:
 65        Random() : _x(123456789),
 66                   _y(362436069),
 67                   _z(521288629),
 68                   _w(88675123) {}
 69
 70        //! `[0, 1]` の乱数を返す(実数)
 71        double random() { return (double)(_xor128()) / 0xFFFFFFFF; }
 72
 73        //! `[0, end]` の乱数を返す(整数)
 74        int randint(const int end) {
 75            assert(0 <= end);
 76            return (((unsigned long long)_xor128() * (end+1)) >> 32);
 77        }
 78
 79        //! `[begin, end]` の乱数を返す(整数)
 80        int randint(const int begin, const int end) {
 81            assert(begin <= end);
 82            return begin + (((unsigned long long)_xor128() * (end-begin+1)) >> 32);
 83        }
 84
 85        //! `[0, end)` の乱数を返す(整数)
 86        int randrange(const int end) {
 87            assert(0 < end);
 88            return (((unsigned long long)_xor128() * end) >> 32);
 89        }
 90
 91        //! `[begin, end)` の乱数を返す(整数)
 92        int randrange(const int begin, const int end) {
 93            assert(begin < end);
 94            return begin + (((unsigned long long)_xor128() * (end-begin)) >> 32);
 95        }
 96
 97        //! `[0, u64_MAX)` の乱数を返す / zobrist hash等の使用を想定
 98        unsigned long long rand_u64() {
 99            return ((unsigned long long)_xor128() << 32) | _xor128();
100        }
101
102        //! `[0, end)` の異なる乱数を2つ返す
103        pair<int, int> rand_pair(const int end) {
104            assert(end >= 2);
105            int u = randrange(end);
106            int v = u + randrange(1, end);
107            if (v >= end) v -= end;
108            if (u > v) swap(u, v);
109            return {u, v};
110        }
111
112        //! `[begin, end)` の異なる乱数を2つ返す
113        pair<int, int> rand_pair(const int begin, const int end) {
114            assert(end - begin >= 2);
115            int u = randrange(begin, end);
116            int v = (u + randrange(1, end-begin));
117            if (v >= end) v -= (end-begin);
118            if (u > v) swap(u, v);
119            return {u, v};
120        }
121
122        //! Note `a`は非const
123        vector<int> rand_vec(const int cnt, vector<int> &a) {
124            int n = a.size();
125            for (int i = 0; i < cnt; ++i) {
126                int j = randrange(i, n);
127                swap(a[i], a[j]);
128            }
129            vector<int> res(a.begin(), a.begin()+cnt);
130            return res;
131        }
132
133        //! `[begin, end)` の乱数を返す(実数)
134        double randdouble(const double begin, const double end) {
135            assert(begin < end);
136            return begin + random() * (end-begin);
137        }
138
139        //! `vector` をインプレースにシャッフルする / `O(n)`
140        template <typename T>
141        void shuffle(vector<T> &a) {
142            int n = a.size();
143            for (int i = 0; i < n-1; ++i) {
144                int j = randrange(i, n);
145                swap(a[i], a[j]);
146            }
147        }
148
149        template <typename T>
150        vector<T> choices(const vector<T> &a, const int k) {
151            assert(a.size() >= k);
152            vector<T> r(k);
153            unordered_set<int> seen;
154            for (int i = 0; i < k; ++i) {
155                int x = randrange(a.size());
156                while (seen.find(x) != seen.end()) x = randrange(a.size());
157                seen.insert(x);
158                r[i] = a[x];
159            }
160            return r;
161        }
162
163        template <typename T>
164        T choice(const vector<T> &a) {
165            return a[randrange(a.size())];
166        }
167
168        template <typename T>
169        T choice(const vector<T> &a, const vector<int> &w, bool normal) {
170            assert(normal == false);
171            assert(a.size() == w.size());
172            double sum = 0.0;
173            for (const int &x: w) sum += x;
174            assert(sum > 0);
175            vector<double> x(w.size());
176            for (int i = 0; i < x.size(); ++i) {
177                x[i] = (double)w[i] / sum;
178                if (i-1 >= 0) x[i] += x[i-1];
179            }
180            return choice(a, x);
181        }
182
183        template <typename T>
184        T choice(const vector<T> &a, const vector<double> &w) {
185            double i = random();
186            int l = -1, r = a.size()-1;
187            while (r - l > 1) {
188                int mid = (l + r) / 2;
189                if (w[mid] <= i) l = mid;
190                else r = mid;
191            }
192            return a[r];
193        }
194    };

308    using ScoreType = double;
309
310    struct Changed {
311        int type;
312        ScoreType pre_score;
313        Changed() {}
314    };
315
316    Changed changed;
317
318    class State {
319      public:
320        ScoreType score;
321        State() {}
322
323        void init() {}
324
325        ScoreType get_score() const { return score; }
326        ScoreType get_true_score() const { return score; }
327
328        // thresholdを超えたら、だめ
329        void modify(const ScoreType threshold) {}
330
331        void rollback() {}
332
333        void advance() {}
334
335        void print() const {}
336    };
337
338    // TIME_LIMIT: ms
339    State sa_run(const double TIME_LIMIT, const bool verbose = false) {
340        titan23::Timer sa_timer;
341
342        // const double START_TEMP = param.start_temp;
343        // const double END_TEMP   = param.end_temp;
344        const double START_TEMP = 1000;
345        const double END_TEMP   = 1;
346        const double TEMP_VAL = (START_TEMP - END_TEMP) / TIME_LIMIT;
347
348        State ans;
349        ans.init();
350        State best_ans = ans;
351        ScoreType score = ans.get_score();
352        ScoreType best_score = score;
353        double now_time;
354
355        int cnt = 0;
356        int upd_cnt = 0;
357        while (true) {
358            // if ((cnt & 31) == 0) now_time = sa_timer.elapsed();
359            now_time = sa_timer.elapsed();
360            if (now_time > TIME_LIMIT) break;
361            ++cnt;
362            ScoreType threshold = score - (START_TEMP-TEMP_VAL*now_time) * log(sa_random.random());
363            changed.pre_score = ans.score;
364            ans.modify(threshold);
365            ScoreType new_score = ans.get_score();
366            if (new_score <= threshold) {
367                ++upd_cnt;
368                ans.advance();
369                score = new_score;
370                if (score < best_score) {
371                    best_score = score;
372                    best_ans = ans;
373                    if (verbose) {
374                        cerr << "Info: score=" << best_score << endl;
375                    }
376                }
377            } else {
378                ans.score = changed.pre_score;
379                ans.rollback();
380            }
381        }
382        if (verbose) {
383            cerr << "Info: upd=" << upd_cnt << endl;
384            cerr << "Info: cnt=" << cnt << endl;
385        }
386        return best_ans;
387    }
388}

 55    template<typename StateType>
 56    StateType sa_run(const double TIME_LIMIT) {
 57        titan23::Timer sa_timer;
 58
 59        // const double START_TEMP = param.start_temp;
 60        // const double END_TEMP   = param.end_temp;
 61        const double START_TEMP = 1000;
 62        const double END_TEMP   = 1;
 63        const double TEMP_VAL = (START_TEMP - END_TEMP) / TIME_LIMIT;
 64
 65        StateType ans;
 66        ans.init();
 67        StateType best_ans = ans;
 68        ScoreType score = ans.get_score();
 69        ScoreType best_score = score;
 70        double now_time;
 71
 72        int cnt = 0;
 73        int upd_cnt = 0;
 74        while (true) {
 75            // if ((cnt & 31) == 0) now_time = sa_timer.elapsed();
 76            now_time = sa_timer.elapsed();
 77            if (now_time > TIME_LIMIT) break;
 78            ++cnt;
 79            ScoreType threshold = score - (START_TEMP-TEMP_VAL*now_time) * log(sa_random.random());
 80            changed.pre_score = ans.score;
 81            ans.modify(threshold);
 82            ScoreType new_score = ans.get_score();
 83            if (new_score <= threshold) {
 84                ++upd_cnt;
 85                ans.advance();
 86                score = new_score;
 87                if (score < best_score) {
 88                    best_score = score;
 89                    best_ans = ans;
 90                    cerr << "score=" << best_score << endl;
 91                }
 92            } else {
 93                ans.score = changed.pre_score;
 94                ans.rollback();
 95            }
 96        }
 97        cerr << "upd=" << upd_cnt << endl;
 98        cerr << "cnt=" << cnt << endl;
 99        return best_ans;
100    }

 25      int balance() const {
 26        int hl = left ? left->height : 0;

 32        size = 1 + (left ? left->size : 0) + (right ? right->size : 0);

 41    AVLTreeSetNodePtr build(vector<T> a) {
 42      auto _build = [&] (auto &&_build, int l, int r) -> AVLTreeSetNodePtr {
 43        int mid = (l + r) / 2;
 44        AVLTreeSetNodePtr node = new AVLTreeSetNode(a[mid]);
 45        if (l != mid) {
 46          node->left = _build(_build, l, mid);
 47          node->left->par = node;
 48        }
 49        if (mid+1 != r) {
 50          node->right = _build(_build, mid+1, r);
 51          node->right->par = node;
 52        }
 53        node->update();
 54        return node;
 55      };
 56
 57      if (a.empty()) return nullptr;
 58      int n = a.size();
 59      bool is_sorted = true;
 60      for (int i = 0; i < n-1; ++i) {
 61        if (!(a[i] < a[i+1])) {
 62          is_sorted = false;
 63          break;
 64        }
 65      }
 66      if (!is_sorted) {
 67        sort(a.begin(), a.end());
 68        a.erase(unique(a.begin(), a.end()), a.end());
 69      }
 70      return _build(_build, 0, a.size());

 73    void _remove_balance(AVLTreeSetNodePtr node) {
 74      while (node) {
 75        AVLTreeSetNodePtr new_node = nullptr;
 76        node->update();
 77        if (node->balance() == 2) {
 78          new_node = node->left->balance() == -1 ? BBSTNode<AVLTreeSetNodePtr>::rotate_LR(node) : BBSTNode<AVLTreeSetNodePtr>::rotate_right(node);
 79        } else if (node->balance() == -2) {
 80          new_node = node->right->balance() == 1 ? BBSTNode<AVLTreeSetNodePtr>::rotate_RL(node) : BBSTNode<AVLTreeSetNodePtr>::rotate_left(node);
 81        } else if (node->balance() != 0) {
 82          node = node->par;
 83          break;
 84        }
 85        if (!new_node) {
 86          node = node->par;
 87          continue;
 88        }
 89        if (!new_node->par) {
 90          this->root = new_node;
 91          return;
 92        }
 93        node = new_node->par;
 94        if (new_node->key < node->key) {
 95          node->left = new_node;
 96        } else {
 97          node->right = new_node;
 98        }
 99        if (new_node->balance() != 0) break;
100      }
101      while (node) {
102        node->update();
103        node = node->par;
104      }
105    }

108      AVLTreeSetNodePtr new_node = nullptr;

110        node->update();
111        if (node->balance() == 0) {
112          node = node->par;
113          break;
114        }
115        if (node->balance() == 2) {
116          new_node = node->left->balance() == -1 ? BBSTNode<AVLTreeSetNodePtr>::rotate_LR(node) : BBSTNode<AVLTreeSetNodePtr>::rotate_right(node);
117          break;
118        } else if (node->balance() == -2) {
119          new_node = node->right->balance() == 1 ? BBSTNode<AVLTreeSetNodePtr>::rotate_RL(node) : BBSTNode<AVLTreeSetNodePtr>::rotate_left(node);
120          break;
121        }
122        node = node->par;
123      }
124      if (new_node) {
125        node = new_node->par;
126        if (node) {

148    bool add(const T &key) {
149      if (!root) {
150        root = new AVLTreeSetNode(key);
151        return true;
152      }
153      AVLTreeSetNodePtr pnode = nullptr;
154      AVLTreeSetNodePtr node = root;
155      while (node) {
156        if (key == node->key) return false;
157        pnode = node;
158        node = key < node->key ? node->left : node->right;
159      }
160      if (key < pnode->key) {
161        pnode->left = new AVLTreeSetNode(key);
162        pnode->left->par = pnode;
163      } else {
164        pnode->right = new AVLTreeSetNode(key);
165        pnode->right->par = pnode;
166      }
167      _add_balance(pnode);
168      return true;

171    void remove_iter(AVLTreeSetNodePtr node) {
172      AVLTreeSetNodePtr pnode = node->par;
173      if (node->left && node->right) {
174        pnode = node;
175        AVLTreeSetNodePtr mnode = node->left;
176        while (mnode->right) {
177          pnode = mnode;
178          mnode = mnode->right;
179        }
180        node->key = mnode->key;
181        node = mnode;
182      }
183      AVLTreeSetNodePtr cnode = (!node->left) ? node->right : node->left;
184      if (cnode) {
185        cnode->par = pnode;
186      }
187      if (pnode) {
188        if (node->key <= pnode->key) {
189          pnode->left = cnode;
190        } else {
191          pnode->right = cnode;
192        }
193        _remove_balance(pnode);
194      } else {
195        root = cnode;
196      }

322      AVLTreeSetNodePtr node = root;

325        if (t == k) return node;

384    void check() const {
385      if (!root) {
386        cout << "height=0" << endl;;
387        cout << "check ok empty." << endl;;
388        return;
389      }
390      cout << "height=" << root->height << endl;
391
392      auto dfs = [&] (auto &&dfs, AVLTreeSetNodePtr node) -> void {
393        int h = 0;
394        int b = 0;
395        int s = 1;
396        if (node->left) {
397          assert(node->left->par == node);
398          assert(node->key > node->left->key);
399          dfs(dfs, node->left);
400          h = max(h, node->left->height);
401          b += node->left->height;
402          s += node->left->size;
403        }
404        if (node->right) {
405          assert(node->right->par == node);
406          assert(node->key < node->right->key);
407          dfs(dfs, node->right);
408          h = max(h, node->right->height);
409          b -= node->right->height;
410          s += node->right->size;
411        }
412        assert(node->height == h+1);
413        assert(-1 <= b && b <= 1);
414        assert(node->size == s);
415      };
416      dfs(dfs, root);
417      cout << "check ok." << endl;

  32  void _build(const vector<uint8_t> &a) {
  33    auto rec = [&] (auto &&rec, Node l, Node r) -> pair<Node, char> {
  34      Node mid = (l + r) >> 1;
  35      char hl = 0, hr = 0;
  36      if (l != mid) {
  37        tie(_left[mid], hl) = rec(rec, l, mid);
  38        _size[mid] += _size[_left[mid]];
  39        _total[mid] += _total[_left[mid]];
  40      }
  41      if (mid + 1 != r) {
  42        tie(_right[mid], hr) = rec(rec, mid+1, r);
  43        _size[mid] += _size[_right[mid]];
  44        _total[mid] += _total[_right[mid]];
  45      }
  46      _balance[mid] = hl - hr;
  47      return {mid, (max(hl, hr)+1)};
  48    };
  49
  50    const int n = a.size();
  51    reserve(n);
  52    Node pre_end = _end;
  53    int indx = _end;
  54    for (int i = 0; i < n; i += _W) {
  55      int j = 0;
  56      int pop = 0;
  57      uint128 v = 0;
  58      while (j < _W && i + j < n) {
  59        v <<= 1;
  60        if (a[i+j]) {
  61          v |= a[i+j];
  62          ++pop;
  63        }
  64        j++;
  65      }
  66      _key[indx] = v;
  67      _bit_len[indx] = j;
  68      _size[indx] = j;
  69      _total[indx] = pop;
  70      ++indx;
  71    }
  72    this->_end = indx;
  73    this->_root = rec(rec, pre_end, _end).first;
  74  }

  93      _balance[u] = -1;
  94      _balance[node] = 1;

 132    Node B = _left[node];
 133    Node E = _right[B];

 136    _size[B] -= _size[_right[E]] + _bit_len[E];
 137    _total[E] = _total[node];
 138    _total[node] -= _total[B] - _total[_right[E]];
 139    _total[B] -= _total[_right[E]] + _popcount(_key[E]);
 140    _right[B] = _left[E];
 141    _left[E] = B;
 142    _left[node] = _right[E];
 143    _right[E] = node;
 144    _update_balance(E);

 165  int _pref(int r) const {
 166    Node node = _root;
 167    int s = 0;
 168    while (r > 0) {
 169      int t = _size[_left[node]] + _bit_len[node];
 170      if (t - _bit_len[node] < r && r <= t) {
 171        r -= _size[_left[node]];
 172        s += _total[_left[node]] + _popcount(_key[node] >> (_bit_len[node] - r));
 173        break;
 174      }
 175      if (t > r) {
 176        node = _left[node];
 177      } else {
 178        s += _total[_left[node]] + _popcount(_key[node]);
 179        node = _right[node];
 180        r -= t;
 181      }
 182    }
 183    return s;
 184  }

 212  void _pop_under(stack<Node> &path, int d, Node node, int res) {
 213    int fd = 0, lmax_total = 0;
 214    char lmax_bit_len = 0;
 215    if (_left[node] && _right[node]) {
 216      path.emplace(node);
 217      d = (d << 1) | 1;
 218      Node lmax = _left[node];
 219      while (_right[lmax]) {
 220        path.emplace(lmax);
 221        d <<= 1;
 222        fd = (fd << 1) | 1;
 223        lmax = _right[lmax];
 224      }
 225      lmax_total = _popcount(_key[lmax]);
 226      lmax_bit_len = _bit_len[lmax];
 227      _key[node] = _key[lmax];
 228      _bit_len[node] = lmax_bit_len;
 229      node = lmax;
 230    }
 231    Node cnode = _left[node] == 0 ? _right[node] : _left[node];
 232    if (!path.empty()) {
 233      ((d & 1) ? _left[path.top()] : _right[path.top()]) = cnode;
 234    } else {
 235      _root = cnode;
 236      return;
 237    }
 238    while (!path.empty()) {
 239      Node new_node = 0;
 240      node = path.top(); path.pop();
 241      _balance[node] -= (d & 1) ? 1 : -1;
 242      _size[node] -= (fd & 1) ? lmax_bit_len : 1;
 243      _total[node] -= (fd & 1) ? lmax_total : res;
 244      d >>= 1;
 245      fd >>= 1;
 246      if (_balance[node] == 2) {
 247        new_node = _balance[_left[node]] < 0 ? _rotate_LR(node) : _rotate_L(node);
 248      } else if (_balance[node] == -2) {
 249        new_node = _balance[_right[node]] > 0 ? _rotate_RL(node) : _rotate_R(node);
 250      } else if (_balance[node] != 0) {
 251        break;
 252      }
 253      if (new_node) {
 254        if (path.empty()) {
 255          _root = new_node;
 256          return;
 257        }
 258        ((d & 1) ? _left[path.top()] : _right[path.top()]) = new_node;
 259        if (_balance[new_node] != 0) break;
 260      }
 261    }
 262    while (!path.empty()) {
 263      node = path.top(); path.pop();
 264      _size[node] -= (fd & 1) ? lmax_bit_len : 1;
 265      _total[node] -= (fd & 1) ? lmax_total : res;
 266      fd >>= 1;
 267    }
 268  }

 313  void insert(int k, bool key) {
 314    if (!_root) {
 315      Node new_node = _make_node(key, 1);
 316      _root = new_node;
 317      return;
 318    }
 319    Node node = _root;
 320    int d = 0;
 321    stack<Node> path;
 322    while (node) {
 323      int t = _size[_left[node]] + _bit_len[node];
 324      if (t - _bit_len[node] <= k && k <= t) break;
 325      d <<= 1;
 326      _size[node]++;
 327      _total[node] += key;
 328      path.emplace(node);
 329      node = (t > k) ? _left[node] : _right[node];
 330      if (t > k) d |= 1;
 331      else k -= t;
 332    }
 333    k -= _size[_left[node]];
 334    if (_bit_len[node] < _W) {
 335      uint128 v = _key[node];
 336      char bl = _bit_len[node] - k;
 337      _key[node] = _bit_insert(v, bl, key);
 338      _bit_len[node]++;
 339      _size[node]++;
 340      _total[node] += key;
 341      return;
 342    }
 343    path.emplace(node);
 344    _size[node]++;
 345    _total[node] += key;
 346    uint128 v = _key[node];
 347    char bl = _W - k;
 348    v = _bit_insert(v, bl, key);
 349    uint128 left_key = v >> _W;
 350    char left_key_popcount = left_key & 1;
 351    _key[node] = v & (((uint128)1 << _W) - 1);
 352    node = _left[node];
 353    d = (d << 1) | 1;
 354    if (!node) {
 355      if (_bit_len[path.top()] < _W) {
 356        _bit_len[path.top()]++;
 357        _key[path.top()] = (_key[path.top()] << 1) | left_key;
 358        return;
 359      } else {
 360        Node new_node = _make_node(left_key, 1);
 361        _left[path.top()] = new_node;
 362      }
 363    } else {
 364      path.emplace(node);
 365      _size[node]++;
 366      _total[node] += left_key_popcount;
 367      d <<= 1;
 368      while (_right[node]) {
 369        node = _right[node];
 370        path.emplace(node);
 371        _size[node]++;
 372        _total[node] += left_key_popcount;
 373        d <<= 1;
 374      }
 375      if (_bit_len[node] < _W) {
 376        _bit_len[node]++;
 377        _key[node] = (_key[node] << 1) | left_key;
 378        return;
 379      } else {
 380        Node new_node = _make_node(left_key, 1);
 381        _right[node] = new_node;
 382      }
 383    }
 384    Node new_node = 0;
 385    while (!path.empty()) {
 386      node = path.top(); path.pop();
 387      _balance[node] += (d & 1) ? 1 : -1;
 388      d >>= 1;
 389      if (_balance[node] == 0) break;
 390      if (_balance[node] == 2) {
 391        new_node = _balance[_left[node]] == -1 ? _rotate_LR(node) : _rotate_L(node);
 392        break;
 393      } else if (_balance[node] == -2) {
 394        new_node = _balance[_right[node]] == 1 ? _rotate_RL(node) : _rotate_R(node);
 395        break;
 396      }
 397    }
 398    if (new_node) {
 399      if (!path.empty()) {
 400        if (d & 1) {
 401          _left[path.top()] = new_node;
 402        } else {
 403          _right[path.top()] = new_node;
 404        }
 405      } else {
 406        _root = new_node;
 407      }
 408    }
 409  }

 411  bool pop(int k) {
 412    Node node = _root;
 413    int d = 0;
 414    stack<Node> path;
 415    while (node) {
 416      int t = _size[_left[node]] + _bit_len[node];
 417      if (t - _bit_len[node] <= k && k < t) break;
 418      path.emplace(node);
 419      node = t > k ? _left[node] : _right[node];
 420      d <<= 1;
 421      if (t > k) d |= 1;
 422      else k -= t;
 423    }
 424    k -= _size[_left[node]];
 425    uint128 v = _key[node];
 426    bool res = (v >> (_bit_len[node] - k - 1)) & 1;
 427    if (_bit_len[node] == 1) {
 428      _pop_under(path, d, node, res);
 429      return res;
 430    }
 431    _key[node] = _bit_pop(v, _bit_len[node]-k);
 432    --_bit_len[node];
 433    --_size[node];
 434    _total[node] -= res;
 435    while (!path.empty()) {
 436      node = path.top(); path.pop();
 437      --_size[node];
 438      _total[node] -= res;
 439    }
 440    return res;
 441  }

 443  void set(int k, bool v) {
 444    Node node = _root;
 445    stack<Node> path;
 446    while (true) {
 447      int t = _size[_left[node]] + _bit_len[node];
 448      path.emplace(node);
 449      if (t - _bit_len[node] <= k && k < t) {
 450        k -= _size[_left[node]];
 451        if (v) {
 452          _key[node] |= (uint128)1 << k;
 453        } else {
 454          _key[node] &= ~((uint128)1 << k);
 455        }
 456        break;
 457      }
 458      if (t > k) {
 459        node = _left[node];
 460      } else {
 461        node = _right[node];
 462        k -= t;
 463      }
 464    }
 465    while (!path.empty()) {
 466      node = path.top(); path.pop();
 467      _total[node] = _popcount(_key[node]) + _total[_left[node]] + _total[_right[node]];
 468    }

 516  int select0(int k) const {
 517    Node node = _root;
 518    int s = 0;
 519    while (true) {
 520      int t = _size[_left[node]] - _total[_left[node]];
 521      if (k < t) {
 522        node = _left[node];
 523      } else if (k >= t + _bit_len[node] - _popcount(_key[node])) {
 524        s += _size[_left[node]] + _bit_len[node];
 525        k -= t + _bit_len[node] - _popcount(_key[node]);
 526        node = _right[node];
 527      } else {
 528        k -= t;
 529        char l = 0, r = _bit_len[node];
 530        while (r - l > 1) {
 531          char m = (l + r) >> 1;
 532          if (m - _popcount(_key[node]>>(_bit_len[node]-m)) > k) r = m;
 533          else l = m;
 534        }
 535        s += _size[_left[node]] + l;
 536        break;
 537      }
 538    }
 539    return s;
 540  }

 571  int _insert_and_rank1(int k, bool key) {
 572    if (_root == 0) {
 573      Node new_node = _make_node(key, 1);
 574      _root = new_node;
 575      return 0;
 576    }
 577    Node node = _root;
 578    int s = 0;
 579    stack<Node> path;
 580    int d = 0;
 581    while (node) {
 582      int t = _size[_left[node]] + _bit_len[node];
 583      if (t - _bit_len[node] <= k && k <= t) break;
 584      if (t <= k) {
 585        s += _total[_left[node]] + _popcount(_key[node]);
 586      }
 587      d <<= 1;
 588      _size[node]++;
 589      _total[node] += key;
 590      path.emplace(node);
 591      node = t > k ? _left[node] : _right[node];
 592      if (t > k) d |= 1;
 593      else k -= t;
 594    }
 595    k -= _size[_left[node]];
 596    s += _total[_left[node]] + _popcount(_key[node] >> (_bit_len[node] - k));
 597    if (_bit_len[node] < _W) {
 598      uint128 v = _key[node];
 599      char bl = _bit_len[node] - k;
 600      _key[node] = _bit_insert(v, bl, key);
 601      _bit_len[node]++;
 602      _size[node]++;
 603      _total[node] += key;
 604      return s;
 605    }
 606    path.emplace(node);
 607    _size[node]++;
 608    _total[node] += key;
 609    uint128 v = _key[node];
 610    char bl = _W - k;
 611    v = _bit_insert(v, bl, key);
 612    uint128 left_key = v >> _W;
 613    char left_key_popcount = left_key & 1;
 614    _key[node] = v & (((uint128)1 << _W) - 1);
 615    node = _left[node];
 616    d = d << 1 | 1;
 617    if (!node) {
 618      if (_bit_len[path.top()] < _W) {
 619        _bit_len[path.top()]++;
 620        _key[path.top()] = (_key[path.top()] << 1) | left_key;
 621        return s;
 622      } else {
 623        Node new_node = _make_node(left_key, 1);
 624        _left[path.top()] = new_node;
 625      }
 626    } else {
 627      path.emplace(node);
 628      _size[node]++;
 629      _total[node] += left_key_popcount;
 630      d <<= 1;
 631      while (_right[node]) {
 632        node = _right[node];
 633        path.emplace(node);
 634        _size[node]++;
 635        _total[node] += left_key_popcount;
 636        d <<= 1;
 637      }
 638      if (_bit_len[node] < _W) {
 639        _bit_len[node]++;
 640        _key[node] = (_key[node] << 1) | left_key;
 641        return s;
 642      } else {
 643        Node new_node = _make_node(left_key, 1);
 644        _right[node] = new_node;
 645      }
 646    }
 647    Node new_node = 0;
 648    while (!path.empty()) {
 649      node = path.top(); path.pop();
 650      _balance[node] += (d & 1) ? 1 : -1;
 651      d >>= 1;
 652      if (_balance[node] == 0) break;
 653      if (_balance[node] == 2) {
 654        new_node = _balance[_left[node]] == -1 ? _rotate_LR(node) : _rotate_L(node);
 655        break;
 656      } else if (_balance[node] == -2) {
 657        new_node = _balance[_right[node]] == 1 ? _rotate_RL(node) : _rotate_R(node);
 658        break;
 659      }
 660    }
 661    if (new_node) {
 662      if (!path.empty()) {
 663        ((d & 1) ? _left[path.top()] : _right[path.top()]) = new_node;
 664      } else {
 665        _root = new_node;
 666      }
 667    }
 668    return s;
 669  }
 670
 671  int _access_pop_and_rank1(int k) {
 672    int s = 0, d = 0;
 673    Node node = _root;
 674    stack<Node> path;
 675    while (node) {
 676      int t = _size[_left[node]] + _bit_len[node];
 677      if (t - _bit_len[node] <= k && k < t) break;
 678      if (t <= k) {
 679        s += _total[_left[node]] + _popcount(_key[node]);
 680      }
 681      path.emplace(node);
 682      node = t > k ? _left[node] : _right[node];
 683      d <<= 1;
 684      if (t > k) d |= 1;
 685      else k -= t;
 686    }
 687    k -= _size[_left[node]];
 688    s += _total[_left[node]] + _popcount(_key[node] >> (_bit_len[node] - k));
 689    uint128 v = _key[node];
 690    bool res = v >> (_bit_len[node] - k - 1) & 1;
 691    if (_bit_len[node] == 1) {
 692      _pop_under(path, d, node, res);
 693      return (s << 1) | res;
 694    }
 695    _key[node] = _bit_pop(v, _bit_len[node]-k);
 696    --_bit_len[node];
 697    --_size[node];
 698    _total[node] -= res;
 699    while (!path.empty()) {
 700      node = path.top(); path.pop();
 701      --_size[node];
 702      _total[node] -= res;
 703    }
 704    return (s << 1) | res;
 705  }
 706
 707  pair<bool, int> _access_ans_rank1(int k) const {
 708    Node node = _root;
 709    int s = 0;
 710    bool res;
 711    while (true) {
 712      int t = _size[_left[node]] + _bit_len[node];
 713      if (t - _bit_len[node] <= k && k < t) {
 714        k -= _size[_left[node]];
 715        s += _total[_left[node]] + _popcount(_key[node] >> (_bit_len[node] - k));
 716        res = (_key[node] >> (_bit_len[node] - k - 1)) & 1;
 717        break;
 718      }
 719      if (t > k) {
 720        node = _left[node];
 721      } else {
 722        s += _total[_left[node]] + _popcount(_key[node]);
 723        node = _right[node];
 724        k -= t;
 725      }
 726    }
 727    return make_pair(res, s);
 728  }

 759    /**
 760     * @brief 動的ウェーブレット行列
 761     *
 762     * @note `DynamicWavletTree` の方が平均的には高速かもしれません
 763     *
 764     * @tparam T
 765     */
 766    template<typename T>
 767    class DynamicWaveletMatrix {
 768      private:
 769        T _sigma;
 770        int _log;
 771        vector<AVLTreeBitVector> _v;
 772        vector<int> _mid;
 773        int _size;
 774
 775        int bit_length(const int n) const {
 776            return 32 - __builtin_clz(n);
 777        }
 778
 779        int bit_length(const long long n) const {
 780            return 64 - __builtin_clz(n);
 781        }
 782
 783        int bit_length(const unsigned long long n) const {
 784            return 64 - __builtin_clz(n);
 785        }
 786
 787        void _build(vector<T> a) {
 788            if (a.empty()) return;
 789            vector<uint8_t> v(_size);
 790            for (int bit = _log-1; bit >= 0; --bit) {
 791                vector<T> zero, one;
 792                for (int i = 0; i < _size; ++i) {
 793                    if ((a[i] >> bit) & 1) {
 794                        v[i] = 1;
 795                        one.emplace_back(a[i]);
 796                    } else {
 797                        v[i] = 0;
 798                        zero.emplace_back(a[i]);
 799                    }
 800                }
 801                _mid[bit] = zero.size();
 802                _v[bit] = AVLTreeBitVector(v);
 803                a = zero;
 804                a.insert(a.end(), one.begin(), one.end());
 805            }
 806        }
 807
 808      public:
 809        DynamicWaveletMatrix(const T sigma)
 810                : _sigma(sigma), _log(bit_length(sigma-1)), _v(_log), _mid(_log), _size(0) {
 811        }
 812
 813        DynamicWaveletMatrix(const T sigma, vector<T> &a)
 814                : _sigma(sigma), _log(bit_length(sigma)), _v(_log), _mid(_log), _size(a.size()) {
 815            _build(a);
 816        }
 817
 818        void reserve(const int n) {
 819            for (int i = 0; i < _log; ++i) {
 820                _v[i].reserve(n);
 821            }
 822        }
 823
 824        void insert(int k, T x) {
 825            for (int bit = _log-1; bit >= 0; --bit) {
 826                if ((x >> bit) & 1) {
 827                    int s = _v[bit]._insert_and_rank1(k, 1);
 828                    k = s + _mid[bit];
 829                } else {
 830                    int s = _v[bit]._insert_and_rank1(k, 0);
 831                    k -= s;
 832                    ++_mid[bit];
 833                }
 834            }
 835            _size++;
 836        }
 837
 838        T pop(int k) {
 839            T ans = 0;
 840            for (int bit = _log-1; bit >= 0; --bit) {
 841                int sb = _v[bit]._access_pop_and_rank1(k);
 842                int s = sb >> 1;
 843                if (sb & 1) {
 844                    ans |= (T)1 << bit;
 845                    k = s + _mid[bit];
 846                } else {
 847                    --_mid[bit];
 848                    k -= s;
 849                }
 850            }
 851            _size--;
 852            return ans;
 853        }
 854
 855        void set(int k, T x) {
 856            assert(0 <= k && k < _size);
 857            pop(k);
 858            insert(k, x);
 859        }
 860
 861        int rank(int r, T x) const {
 862            int l = 0;
 863            for (int bit = _log-1; bit >= 0; --bit) {
 864                if ((x >> bit) & 1) {
 865                    l = _v[bit].rank1(l) + _mid[bit];
 866                    r = _v[bit].rank1(r) + _mid[bit];
 867                } else {
 868                    l = _v[bit].rank0(l);
 869                    r = _v[bit].rank0(r);
 870                }
 871            }
 872            return r - l;
 873        }
 874
 875        T access(int k) const {
 876            T s = 0;
 877            for (int bit = _log-1; bit >= 0; --bit) {
 878                if (_v[bit].access(k)) {
 879                    s |= (T)1 << bit;
 880                    k = _v[bit].rank1(k) + _mid[bit];
 881                } else {
 882                    k = _v[bit].rank0(k);
 883                }
 884            }
 885            return s;
 886        }
 887
 888        T kth_smallest(int l, int r, int k) const {
 889            T s = 0;
 890            for (int bit = _log-1; bit >= 0; --bit) {
 891                int l0 = _v[bit].rank0(l);
 892                int r0 = _v[bit].rank0(r);
 893                int cnt = r0 - l0;
 894                if (cnt <= k) {
 895                    s |= (T)1 << bit;
 896                    k -= cnt;
 897                    l = l - l0 + _mid[bit];
 898                    r = r - r0 + _mid[bit];
 899                } else {
 900                    l = l0;
 901                    r = r0;
 902                }
 903            }
 904            return s;
 905        }
 906
 907        T kth_largest(int l, int r, int k) const {
 908            return kth_smallest(l, r, r-l-k-1);
 909        }
 910
 911        vector<pair<T, int>> topk(int l, int r, int k) {
 912            priority_queue<tuple<int, T, int, char>> hq;
 913            hq.emplace(r-l, 0, l, _log-1);
 914            vector<pair<T, int>> ans;
 915            while (!hq.empty()) {
 916                auto [length, x, l, bit] = hq.top();
 917                hq.pop();
 918                if (bit == -1) {
 919                    ans.emplace_back(x, length);
 920                    --k;
 921                    if (k == 0) break;
 922                } else {
 923                    int r = l + length;
 924                    int l0 = _v[bit].rank0(l);
 925                    int r0 = _v[bit].rank0(r);
 926                    if (l0 < r0) hq.emplace(r0-l0, x, l0, bit-1);
 927                    int l1 = _v[bit].rank1(l) + _mid[bit];
 928                    int r1 = _v[bit].rank1(r) + _mid[bit];
 929                    if (l1 < r1) hq.emplace(r1-l1, x|((T)1<<(T)bit), l1, bit-1);
 930                }
 931            }
 932            return ans;
 933        }
 934
 935        int select(int k, T x) const {
 936            T s = 0;
 937            for (int bit = _log-1; bit >= 0; --bit) {
 938                if ((x >> bit) & 1) {
 939                    s = _v[bit].rank0(_size) + _v[bit].rank1(s);
 940                } else {
 941                    s = _v[bit].rank0(s);
 942                }
 943            }
 944            s += k;
 945            for (int bit = 0; bit < _log; ++bit) {
 946                if ((x >> bit) & 1) {
 947                    s = _v[bit].select1(s - _v[bit].rank0(_size));
 948                } else {
 949                    s = _v[bit].select0(s);
 950                }
 951            }
 952            return s;
 953        }
 954
 955        int range_freq(int l, int r, T x) const {
 956            int ans = 0;
 957            for (int bit = _log-1; bit >= 0; --bit) {
 958                int l0 = _v[bit].rank0(l);
 959                int r0 = _v[bit].rank0(r);
 960                if ((x >> bit) & 1) {
 961                    ans += r0 - l0;
 962                    l += _mid[bit] - l0;
 963                    r += _mid[bit] - r0;
 964                } else {
 965                    l = l0;
 966                    r = r0;
 967                }
 968            }
 969            return ans;
 970        }
 971
 972        int range_freq(int l, int r, int x, int y) const {
 973            return range_freq(l, r, y) - range_freq(l, r, x);
 974        }
 975
 976        T prev_value(int l, int r, T x) const {
 977            return kth_smallest(l, r, _range_freq(l, r, x)-1);
 978        }
 979
 980        T next_value(int l, int r, T x) const {
 981            return kth_smallest(l, r, _range_freq(l, r, x));
 982        }
 983
 984        int range_count(int l, int r, T x) const {
 985            return rank(r, x) - rank(l, x);
 986        }
 987
 988        vector<T> tovector() const {
 989
 990            vector<T> a(_size);
 991            for (int i = 0; i < _size; ++i) {
 992                a[i] = access(i);
 993            }
 994            return a;
 995        }

 997        void print() const {
 998            vector<T> a = tovector();
 999            int n = (int)a.size();
1000            cout << "[";
1001            for (int i = 0; i < n-1; ++i) {
1002                cout << a[i] << ", ";
1003            }
1004            if (n > 0) {
1005                cout << a.back();
1006            }
1007            cout << "]";
1008            cout << endl;
1009        }

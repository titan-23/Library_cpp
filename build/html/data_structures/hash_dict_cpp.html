  7namespace titan23 {
  8  template<typename V>
  9  class HashDict {
 10   private:
 11    using u64 = unsigned long long;
 12    static constexpr const u64 k = 0x517cc1b727220a95;
 13    vector<u64> exist;
 14    vector<u64> keys;
 15    vector<V> vals;
 16    int msk, xor_;
 17    int size;
 18
 19    int hash(const u64 key) const {
 20      return (((((key>>32)&msk) ^ (key&msk) ^ xor_)) * (k & msk)) & msk;
 21    }
 22
 23    pair<int, bool> get_pos(const u64 key) const {
 24      int h = hash(key), s = keys.size();
 25      while (true) {
 26        if (!(exist[h>>6]>>(h&63)&1)) return {h, false};
 27        if (keys[h] == key) return {h, true};
 28        ++h;
 29        if (h == s) h = 0;
 30      }
 31    }
 32
 33    int bit_length(const int x) const {
 34      if (x == 0) return 0;
 35      return 32 - __builtin_clz(x);
 36    }
 37
 38    void rebuild() {
 39      vector<u64> old_exist = exist;
 40      vector<u64> old_keys = keys;
 41      vector<V> old_vals = vals;
 42      exist.resize(2*old_exist.size());
 43      fill(exist.begin(), exist.end(), 0);
 44      keys.resize(2*old_keys.size());
 45      vals.resize(2*old_vals.size());
 46      size = 0;
 47      msk = (1<<bit_length(keys.size()-1))-1;
 48      random_device rd;
 49      mt19937 gen(rd());
 50      uniform_int_distribution<int> dis(0, msk);
 51      xor_ = dis(gen);
 52      for (int i = 0; i < old_keys.size(); ++i) {
 53        if (old_exist[i>>6]>>(i&63)&1) {
 54          set(old_keys[i], old_vals[i]);
 55        }
 56      }
 57    }
 58
 59   public:
 60    HashDict() : exist(1, 0), keys(1), vals(1), msk(0), xor_(0), size(0) {}
 61
 62    HashDict(const int n) {
 63      const int s = 1<<bit_length(n);
 64      exist.resize((s>>6)+1, 0);
 65      keys.resize(s);
 66      vals.resize(s);
 67      msk = (1<<bit_length(keys.size()-1))-1;
 68      random_device rd;
 69      mt19937 gen(rd());
 70      uniform_int_distribution<int> dis(0, msk);
 71      xor_ = dis(gen);
 72    }
 73
 74    V get(const u64 key) const {
 75      const auto [pos, exist_res] = get_pos(key);
 76      if (!exist_res) return V();
 77      else return vals[pos];
 78    }
 79
 80    V get(const u64 key, const V missing) const {
 81      const auto [pos, exist_res] = get_pos(key);
 82      if (!exist_res) return missing;
 83      else return vals[pos];
 84    }
 85
 86    bool contains(const u64 key) const {
 87      return get_pos(key).second;
 88    }
 89
 90    V operator[] (const u64 key) const {
 91      return get(key);
 92    }
 93
 94    void set(const u64 key, const V val) {
 95      const auto [pos, is_exist] = get_pos(key);
 96      vals[pos] = val;
 97      if (!is_exist) {
 98        exist[pos>>6] |= 1<<(pos&63);
 99        keys[pos] = key;
100        ++size;
101        if (2*size > keys.size()) {
102          rebuild();
103        }
104      }
105    }
106  };
107}

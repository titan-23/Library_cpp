 35      Node(T key, F lazy) : key(key), data(key), left(nullptr), right(nullptr), lazy(lazy), rev(0), size(1) {}

 37      NodePtr copy() const {
 38        NodePtr node = make_shared<Node>(key, lazy);

 47      double balance() const {
 48        return ((left ? left->size : 0) + 1.0) / (size + 1.0);

 51      void propagate() {
 52        if (rev) {
 53          NodePtr l = (left)? left->copy(): nullptr;
 54          NodePtr r = (right)? right->copy(): nullptr;
 55          left = r;
 56          right = l;
 57          if (left) left->rev ^= 1;
 58          if (right) right->rev ^= 1;
 59          rev = 0;
 60        }
 61        if (lazy != id()) {
 62          if (left) {
 63            left = left->copy();
 64            left->key = mapping(lazy, left->key);
 65            left->data = mapping(lazy, left->data);
 66            left->lazy = composition(lazy, left->lazy);
 67          }
 68          if (right) {
 69            right = right->copy();
 70            right->key = mapping(lazy, right->key);
 71            right->data = mapping(lazy, right->data);
 72            right->lazy = composition(lazy, right->lazy);
 73          }
 74          lazy = id();
 75        }
 76      }

 78      void update() {
 79        size = 1;
 80        data = key;
 81        if (left) {
 82          size += left->size;
 83          data = op(left->data, key);
 84        }
 85        if (right) {
 86          size += right->size;
 87          data = op(data, right->data);
 88        }
 89      }

 92        cout << "this : key=" << key << ", size=" << size << endl;
 93        if (left)  cout << "to-left" << endl;
 94        if (right) cout << "to-right" << endl;

101    void _build(vector<T> const &a) {
102      auto build = [&] (auto &&build, int l, int r) -> NodePtr {
103        int mid = (l + r) >> 1;
104        NodePtr node = make_shared<Node>(a[mid], id());
105        if (l != mid) node->left = build(build, l, mid);
106        if (mid+1 != r) node->right = build(build, mid+1, r);
107        node->update();
108        return node;
109      };
110      root = build(build, 0, (int)a.size());

131    NodePtr _balance_left(NodePtr &node) {
132      node->right->propagate();
133      node->right = node->right->copy();
134      NodePtr u = node->right;
135      if (u->balance() >= BETA) {
136        u->left->propagate();
137        node->right = _rotate_right(u);
138      }
139      u = _rotate_left(node);
140      return u;
141    }

143    NodePtr _balance_right(NodePtr &node) {
144      node->left->propagate();
145      node->left = node->left->copy();
146      NodePtr u = node->left;
147      if (u->balance() <= 1.0 - BETA) {
148        u->right->propagate();
149        node->left = _rotate_left(u);
150      }
151      u = _rotate_right(node);
152      return u;
153    }

155    NodePtr _merge_with_root(NodePtr l, NodePtr root, NodePtr r) {
156      int ls = l? l->size: 0;
157      int rs = r? r->size: 0;
158      double diff = (double)(ls+1.0) / (ls+rs+2.0);
159      if (diff > 1.0-ALPHA) {
160        l->propagate();
161        l = l->copy();
162        l->right = _merge_with_root(l->right, root, r);
163        l->update();
164        if (!(ALPHA <= l->balance() && l->balance() <= 1.0-ALPHA)) {
165          return _balance_left(l);
166        }
167        return l;
168      }
169      if (diff < ALPHA) {
170        r->propagate();
171        r = r->copy();
172        r->left = _merge_with_root(l, root, r->left);
173        r->update();
174        if (!(ALPHA <= r->balance() && r->balance() <= 1.0-ALPHA)) {
175          return _balance_right(r);
176        }
177        return r;
178      }
179      root = root->copy();
180      root->left = l;
181      root->right = r;
182      root->update();
183      return root;
184    }

186    pair<NodePtr, NodePtr> _pop_right(NodePtr &node) {
187      vector<NodePtr> path;
188      node->propagate();
189      node = node->copy();
190      NodePtr mx = node;
191      while (node->right) {
192        path.emplace_back(node);
193        node->right->propagate();
194        node = node->right->copy();
195        mx = node;
196      }
197      path.emplace_back(node->left? node->left->copy(): nullptr);
198      while ((int)path.size() > 1) {
199        node = path.back();
200        path.pop_back();
201        if (!node) {
202          path.back()->right = nullptr;
203          path.back()->update();
204          continue;
205        }
206        double b = node->balance();
207        if (ALPHA <= b && b <= 1.0-ALPHA) {
208          path.back()->right = node;
209        } else if (b > 1.0-ALPHA) {
210          path.back()->right = _balance_right(node);
211        } else {
212          path.back()->right = _balance_left(node);
213        }
214        path.back()->update();
215      }
216      if (path[0]) {
217        double b = path[0]->balance();
218        if (b > 1.0-ALPHA) {
219          path[0] = _balance_right(path[0]);
220        } else if (b < ALPHA) {
221          path[0] = _balance_left(path[0]);
222        }
223      }
224      mx->left = nullptr;
225      mx->update();
226      return {path[0], mx};
227    }

239    pair<NodePtr, NodePtr> _split_node(NodePtr &node, int k) {
240      if (!node) {return {nullptr, nullptr};}
241      node->propagate();
242      int tmp = node->left? k-node->left->size: k;
243      if (tmp == 0) {
244        return {node->left, _merge_with_root(nullptr, node, node->right)};
245      } else if (tmp < 0) {
246        auto [l, r] = _split_node(node->left, k);
247        return {l, _merge_with_root(r, node, node->right)};
248      } else {
249        auto [l, r] = _split_node(node->right, tmp-1);
250        return {_merge_with_root(node->left, node, l), r};
251      }

278    MyPersistentLazyWBTree apply(int l, int r, F f) {
279      if (l >= r) return _new(this->root ? root->copy() : nullptr);
280      auto dfs = [&] (auto &&dfs, NodePtr node, int left, int right) -> NodePtr {
281        if (right <= l || r <= left) return node;
282        node->propagate();
283        NodePtr nnode = node->copy();
284        if (l <= left && right < r) {
285          nnode->key = mapping(f, nnode->key);
286          nnode->data = mapping(f, nnode->data);
287          nnode->lazy = composition(f, nnode->lazy);
288          return nnode;
289        }
290        int lsize = nnode->left ? nnode->left->size : 0;
291        if (nnode->left) nnode->left = dfs(dfs, nnode->left, left, left+lsize);
292        if (l <= left+lsize && left+lsize < r) nnode->key = mapping(f, nnode->key);;
293        if (nnode->right) nnode->right = dfs(dfs, nnode->right, left+lsize+1, right);
294        nnode->update();
295        return nnode;
296      };
297      return _new(dfs(dfs, root, 0, len()));
298    }

300    T prod(int l, int r) {
301      if (l >= r) return e();
302      auto dfs = [&] (auto &&dfs, NodePtr node, int left, int right) -> T {
303        if (right <= l || r <= left) return e();
304        node->propagate();
305        if (l <= left && right < r) return node->data;
306        int lsize = node->left ? node->left->size : 0;
307        T res = e();
308        if (node->left) res = dfs(dfs, node->left, left, left+lsize);
309        if (l <= left+lsize && left+lsize < r) res = op(res, node->key);
310        if (node->right) res = op(res, dfs(dfs, node->right, left+lsize+1, right));
311        return res;
312      };
313      return dfs(dfs, root, 0, len());
314    }

332      assert(0 <= l && l <= r && r <= len());
333      if (l >= r) return _new(root? root->copy(): nullptr);
334      auto [s_, t] = _split_node(root, r);
335      auto [u, s] = _split_node(s_, l);
336      s->rev ^= 1;
337      NodePtr root = _merge_node(_merge_node(u, s), t);

342      NodePtr node = root;
343      vector<NodePtr> stack;
344      vector<T> a;
345      a.reserve(len());
346      while ((!stack.empty()) || node) {
347        if (node) {
348          node->propagate();
349          stack.emplace_back(node);
350          node = node->left;
351        } else {
352          node = stack.back();
353          stack.pop_back();
354          a.emplace_back(node->key);
355          node = node->right;
356        }
357      }

365    MyPersistentLazyWBTree set(int k, T v) {
366      if (k < 0) k += len();
367      assert(0 <= k && k < len());
368      NodePtr node = root->copy();
369      NodePtr root = node;
370      NodePtr pnode = nullptr;
371      int d = 0;
372      vector<NodePtr> path = {node};
373      while (1) {
374        node->propagate();
375        int t = node->left? node->left->size: 0;
376        if (t == k) {
377          node = node->copy();
378          node->key = v;
379          path.emplace_back(node);
380          if (d) pnode->left = node;
381          else pnode->right = node;
382          while (!path.empty()) {
383            update(path.back());
384            path.pop_back();
385          }
386          return _new(root);
387        }
388        pnode = node;
389        if (t < k) {
390          k -= t + 1;
391          node = node->right->copy();
392          d = 0;
393        } else {
394          d = 1;
395          node = node->left->copy();
396        }
397        path.emplace_back(node);
398        if (d) pnode->left = node;
399        else pnode->right = node;
400      }
401    }

404      if (k < 0) k += len();

413        if (t < k) {

415          node = node->right;
416        } else {
417          node = node->left;
418        }
419      }

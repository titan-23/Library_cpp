 44        void _build(vector<T> const &a) {
 45            if (a.empty()) {
 46                root = nullptr;
 47                return;
 48            }
 49            auto build = [&] (auto &&build, int l, int r) -> NodePtr {
 50                int mid = (l + r) >> 1;
 51                NodePtr node = new Node(a[mid]);
 52                if (l != mid) node->left = build(build, l, mid);
 53                if (mid+1 != r) node->right = build(build, mid+1, r);
 54                node->_update();
 55                return node;
 56            };
 57            root = build(build, 0, (int)a.size());

 61            NodePtr u = node->left;
 62            node->left = u->right;
 63            u->right = node;

 70            NodePtr u = node->right;

 78        NodePtr _balance_left(NodePtr node) {
 79            NodePtr u = node->right;
 80            if (u->balance() >= BETA) {
 81                node->right = _rotate_right(u);
 82            }
 83            return _rotate_left(node);
 84        }

 86        NodePtr _balance_right(NodePtr node) {
 87            NodePtr u = node->left;
 88            if (u->balance() <= 1.0 - BETA) {
 89                node->left = _rotate_left(u);
 90            }
 91            return _rotate_right(node);
 92        }

 94        NodePtr _merge_with_root(NodePtr l, NodePtr root, NodePtr r) {
 95            int ls = l ? l->size: 0;
 96            int rs = r ? r->size: 0;
 97            double diff = (double)(ls+1.0) / (ls+rs+2.0);
 98            if (diff > 1.0-ALPHA) {
 99                l->right = _merge_with_root(l->right, root, r);
100                l->_update();
101                if (!(ALPHA <= l->balance() && l->balance() <= 1.0-ALPHA)) {
102                    return _balance_left(l);
103                }
104                return l;
105            }
106            if (diff < ALPHA) {
107                r->left = _merge_with_root(l, root, r->left);
108                r->_update();
109                if (!(ALPHA <= r->balance() && r->balance() <= 1.0-ALPHA)) {
110                    return _balance_right(r);
111                }
112                return r;
113            }
114            root->left = l;
115            root->right = r;
116            root->_update();
117            return root;
118        }

120        pair<NodePtr, NodePtr> _pop_right(NodePtr node) {
121            vector<NodePtr> path;
122            NodePtr mx = node;
123            while (node->right) {
124                path.emplace_back(node);
125                node = node->right;
126                mx = node;
127            }
128            path.emplace_back(node->left? node->left: nullptr);
129            while ((int)path.size() > 1) {
130                    node = path.back();
131                    path.pop_back();
132                    if (!node) {
133                    path.back()->right = nullptr;
134                    path.back()->_update();
135                    continue;
136                }
137                double b = node->balance();
138                if (ALPHA <= b && b <= 1.0-ALPHA) {
139                    path.back()->right = node;
140                } else if (b > 1.0-ALPHA) {
141                    path.back()->right = _balance_right(node);
142                } else {
143                    path.back()->right = _balance_left(node);
144                }
145                path.back()->_update();
146            }
147            if (path[0]) {
148                double b = path[0]->balance();
149                if (b > 1.0-ALPHA) {
150                    path[0] = _balance_right(path[0]);
151                } else if (b < ALPHA) {
152                    path[0] = _balance_left(path[0]);
153                }
154            }
155            mx->left = nullptr;
156            mx->_update();
157            return {path[0], mx};
158        }

168        pair<NodePtr, NodePtr> _split_node(NodePtr node, int k) {
169            if (!node) return {nullptr, nullptr};
170            int tmp = node->left? k-node->left->size: k;
171            NodePtr nl = node->left;
172            NodePtr nr = node->right;
173            if (tmp == 0) {
174                return {nl, _merge_with_root(nullptr, node, nr)};
175            } else if (tmp < 0) {
176                auto [l, r] = _split_node(nl, k);
177                return {l, _merge_with_root(r, node, nr)};
178            } else {
179                auto [l, r] = _split_node(nr, tmp-1);
180                return {_merge_with_root(node->left, node, l), r};
181            }

235        T pop(int k) {
236            if (k < 0) k += len();
237            assert(0 <= k && k < len());
238            vector<NodePtr> path;
239            vector<short> path_d;
240            NodePtr node = this->root;
241            T res;
242            int d = 0;
243            while (1) {
244                int t = node->left? node->left->size: 0;
245                if (t == k) {
246                    res = node->key;
247                    break;
248                }
249                int d = (t < k)? 0: 1;
250                path.emplace_back(node);
251                path_d.emplace_back(d);
252                if (d) {
253                    node = node->left;
254                } else {
255                    k -= t + 1;
256                    node = node->right;
257                }
258            }
259            if (node->left && node->right) {
260                NodePtr lmax = node->left;
261                path.emplace_back(node);
262                path_d.emplace_back(1);
263                while (lmax->right) {
264                    path.emplace_back(lmax);
265                    path_d.emplace_back(0);
266                    lmax = lmax->right;
267                }
268                node->key = lmax->key;
269                node = lmax;
270            }
271            NodePtr cnode = (node->left)? node->left: node->right;
272            if (!path.empty()) {
273                if (path_d.back()) path.back()->left = cnode;
274                else path.back()->right = cnode;
275            } else {
276                this->root = cnode;
277                return res;
278            }
279            while (!path.empty()) {
280                NodePtr new_node = nullptr;
281                node = path.back();
282                node->_update();
283                path.pop_back();
284                path_d.pop_back();
285                double b = node->balance();
286                if (b < ALPHA) {
287                    new_node = _balance_left(node);
288                } else if (b > 1.0-ALPHA) {
289                    new_node = _balance_right(node);
290                }
291                if (new_node) {
292                    if (path.empty()) {
293                        this->root = new_node;
294                        break;
295                    }
296                    if (path_d.back()) {
297                        path.back()->left = new_node;
298                    } else {
299                        path.back()->right = new_node;
300                    }
301                }
302            }
303            return res;
304        }

326            if (k < 0) k += len();
327            assert(0 <= k && k < len());
328            NodePtr node = root;
329            NodePtr root = node;
330            NodePtr pnode = nullptr;
331            int d = 0;
332            vector<NodePtr> path = {node};
333            while (1) {
334                int t = node->left? node->left->size: 0;
335                if (t == k) {
336                    node->key = v;

338                    if (pnode) {
339                        if (d) pnode->left = node;
340                        else pnode->right = node;
341                    }
342                    while (!path.empty()) {
343                        path.back()->_update();
344                        path.pop_back();
345                    }
346                    return;
347                }
348                pnode = node;
349                d = (t < k)? 0: 1;
350                if (d) {
351                    pnode->left = node = node->left;
352                } else {
353                    k -= t + 1;
354                    pnode->right = node = node->right;
355                }
356                path.emplace_back(node);
357            }

414        void isok() {
415            auto rec = [&] (auto &&rec, NodePtr node) -> pair<int, int> {
416                int ls = 0, rs = 0;
417                int height = 0;
418                int h;
419                if (node->left) {
420                pair<int, int> res = rec(rec, node->left);
421                ls = res.first;
422                h = res.second;
423                height = max(height, h);
424                }
425                if (node->right) {
426                pair<int, int> res = rec(rec, node->right);
427                rs = res.first;
428                h = res.second;
429                height = max(height, h);
430                }
431                int s = ls + rs + 1;
432                double b = (double)(ls+1) / (s+1);
433                assert(s == node->size);
434                assert(ALPHA <= b && b <= 1-ALPHA);
435                return {s, height+1};
436            };
437            if (root == nullptr) return;
438            auto [_, h] = rec(rec, root);
439            // printf("height=%d\n", h);

  2#include <stack>
  3using namespace std;
  4
  5// HLD
  6namespace titan23 {
  7
  8    class HLD {
  9      private:
 10        vector<vector<int>> G;
 11        int root, n;
 12        vector<int> size, par, dep, nodein, nodeout, head, hld;
 13
 14        void _dfs() {
 15            dep[root] = 0;
 16            stack<int> st;
 17            st.emplace(root);
 18            while (!st.empty()) {
 19                int v = st.top(); st.pop();
 20                if (v >= 0) {
 21                    int dep_nxt = dep[v] + 1;
 22                    for (const int x: G[v]) {
 23                        if (dep[x] != -1) continue;
 24                        dep[x] = dep_nxt;
 25                        st.emplace(~x);
 26                        st.emplace(x);
 27                    }
 28                } else {
 29                    v = ~v;
 30                    for (int i = 0; i < (int)G[v].size(); ++i) {
 31                        int x = G[v][i];
 32                        if (dep[x] < dep[v]) {
 33                            par[v] = x;
 34                            continue;
 35                        }
 36                        size[v] += size[x];
 37                        if (size[x] > size[G[v][0]]) {
 38                            swap(G[v][0], G[v][i]);
 39                        }
 40                    }
 41                }
 42            }
 43
 44            int curtime = 0;
 45            st.emplace(~root);
 46            st.emplace(root);
 47            while (!st.empty()) {
 48                int v = st.top(); st.pop();
 49                if (v >= 0) {
 50                    if (par[v] == -1) {
 51                        head[v] = v;
 52                    }
 53                    nodein[v] = curtime;
 54                    hld[curtime] = v;
 55                    ++curtime;
 56                    if (G[v].empty()) continue;
 57                    int G_v0 = (int)G[v][0];
 58                    for (int i = (int)G[v].size()-1; i >= 0; --i) {
 59                        int x = G[v][i];
 60                        if (x == par[v]) continue;
 61                        head[x] = (x == G_v0? head[v]: x);
 62                        st.emplace(~x); 
 63                        st.emplace(x); 
 64                    }
 65                } else {
 66                    nodeout[~v] = curtime;          
 67                }
 68            }
 69        }
 70
 71      public:
 72        HLD() {}
 73        HLD(const vector<vector<int>> &G, const int root) :
 74                G(G), root(root), n(G.size()),
 75                size(n, 1), par(n, -1), dep(n, -1),
 76                nodein(n, -1), nodeout(n, -1), head(n, -1), hld(n, -1) {
 77            _dfs();
 78        }
 79
 80        template<typename T>
 81        vector<T> build_list(vector<T> a) const {
 82            vector<T> res(a.size());
 83            for (int i = 0; i < n; ++i) {
 84                res[i] = a[hld[i]];
 85            }
 86            return res;
 87        }
 88
 89        //! `u`, `v` の lca を返す / `O(logn)`
 90        int lca(int u, int v) const {
 91            while (true) {
 92                if (nodein[u] > nodein[v]) swap(u, v);
 93                if (head[u] == head[v]) return u;
 94                v = par[head[v]];
 95            }
 96        }
 97
 98        //! `s` から `t` へ、`k` 個進んだ頂点を返す / `O(logn)`
 99        int path_kth_elm(int s, int t, int k) {
100            int x = lca(s, t);
101            int d = dep[s] + dep[t] - 2*dep[x];
102            if (d < k) return -1;
103            if (dep[s] - dep[x] < k) {
104                s = t;
105                k = d - k;
106            }
107            int hs = head[s];
108            while (dep[s] - dep[hs] < k) {
109                k -= dep[s] - dep[hs] + 1;
110                s = par[hs];
111                hs = head[s];
112            }
113            return hld[nodein[s] - k];
114        }
115    };
116}  // namespace titan23

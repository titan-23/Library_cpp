 11    class HLD {
 12      private:
 13        vector<vector<int>> G;
 14        int root, n;
 15        vector<int> size, par, dep, nodein, nodeout, head, hld;
 16
 17        void _dfs() {
 18            dep[root] = 0;
 19            stack<int> st;
 20            st.emplace(root);
 21            while (!st.empty()) {
 22                int v = st.top(); st.pop();
 23                if (v >= 0) {
 24                    int dep_nxt = dep[v] + 1;
 25                    for (const int x: G[v]) {
 26                        if (dep[x] != -1) continue;
 27                        dep[x] = dep_nxt;
 28                        st.emplace(~x);
 29                        st.emplace(x);
 30                    }
 31                } else {
 32                    v = ~v;
 33                    for (int i = 0; i < (int)G[v].size(); ++i) {
 34                        int x = G[v][i];
 35                        if (dep[x] < dep[v]) {
 36                            par[v] = x;
 37                            continue;
 38                        }
 39                        size[v] += size[x];
 40                        if (size[x] > size[G[v][0]]) {
 41                            swap(G[v][0], G[v][i]);
 42                        }
 43                    }
 44                }
 45            }
 46
 47            int curtime = 0;
 48            st.emplace(~root);
 49            st.emplace(root);
 50            while (!st.empty()) {
 51                int v = st.top(); st.pop();
 52                if (v >= 0) {
 53                    if (par[v] == -1) {
 54                        head[v] = v;
 55                    }
 56                    nodein[v] = curtime;
 57                    hld[curtime] = v;
 58                    ++curtime;
 59                    if (G[v].empty()) continue;
 60                    int G_v0 = (int)G[v][0];
 61                    for (int i = (int)G[v].size()-1; i >= 0; --i) {
 62                        int x = G[v][i];
 63                        if (x == par[v]) continue;
 64                        head[x] = (x == G_v0? head[v]: x);
 65                        st.emplace(~x); 
 66                        st.emplace(x); 
 67                    }
 68                } else {
 69                    nodeout[~v] = curtime;          
 70                }
 71            }
 72        }
 73
 74      public:
 75        HLD() {}
 76        HLD(const vector<vector<int>> &G, const int root) :
 77                G(G), root(root), n(G.size()),
 78                size(n, 1), par(n, -1), dep(n, -1),
 79                nodein(n, -1), nodeout(n, -1), head(n, -1), hld(n, -1) {
 80            _dfs();
 81        }
 82
 83        template<typename T>
 84        vector<T> build_list(vector<T> a) const {
 85            vector<T> res(a.size());
 86            for (int i = 0; i < n; ++i) {
 87                res[i] = a[hld[i]];
 88            }
 89            return res;
 90        }
 91
 92        //! `u`, `v` の lca を返す / `O(logn)`
 93        int lca(int u, int v) const {
 94            while (true) {
 95                if (nodein[u] > nodein[v]) swap(u, v);
 96                if (head[u] == head[v]) return u;
 97                v = par[head[v]];
 98            }
 99        }
100
101        //! `s` から `t` へ、`k` 個進んだ頂点を返す / `O(logn)`
102        int path_kth_elm(int s, int t, int k) {
103            int x = lca(s, t);
104            int d = dep[s] + dep[t] - 2*dep[x];
105            if (d < k) return -1;
106            if (dep[s] - dep[x] < k) {
107                s = t;
108                k = d - k;
109            }
110            int hs = head[s];
111            while (dep[s] - dep[hs] < k) {
112                k -= dep[s] - dep[hs] + 1;
113                s = par[hs];
114                hs = head[s];
115            }
116            return hld[nodein[s] - k];
117        }
118    };

129    template <class T,
130              T (*_op)(T, T),
131              T (*_e)()>
132    struct SegmentTree {
133      private:
134        int n, _size, _log;
135        vector<T> _data;
136
137      public:
138        SegmentTree() {}
139
140        SegmentTree(const int n) {
141            _build(n);
142        }
143
144        SegmentTree(const vector<T> &a) {
145            int n = (int)a.size();
146            _build(n);
147            for (int i = 0; i < n; ++i) {
148                _data[i+_size] = a[i];
149            }
150            for (int i = _size-1; i > 0; --i) {
151                _data[i] = _op(_data[i<<1], _data[i<<1|1]);
152            }
153        }
154
155        void _build(const int n) {
156            this->n = n;
157            this->_log = 32 - __builtin_clz(n);
158            this->_size = 1 << _log;
159            this->_data.resize(_size << 1, _e());
160        }
161
162        T get(int const k) const {
163            return _data[(k<0? (k+n+_size): (k+_size))];
164        }
165
166        void set(int k, const T v) {
167            if (k < 0) k += n;
168            k += _size;
169            _data[k] = v;
170            for (int i = 0; i < _log; ++i) {
171                k >>= 1;
172                _data[k] = _op(_data[k<<1], _data[k<<1|1]);
173            }
174        }
175
176        T prod(int l, int r) const {
177            l += _size;
178            r += _size;
179            T lres = _e(), rres = _e();
180            while (l < r) {
181                if (l & 1) lres = _op(lres, _data[l++]);
182                if (r & 1) rres = _op(_data[r^1], rres);
183                l >>= 1;
184                r >>= 1;
185            }
186            return _op(lres, rres);
187        }
188
189        T all_prod() const {
190            return _data[1];
191        }
192
193        template<typename F>  // F: function<bool (T)> f
194        int max_right(int l, F &&f) const {
195            assert(0 <= l && l <= _size);
196            // assert(f(_e()));
197            if (l == n) return n;
198            l += _size;
199            T s = _e();
200            while (1) {
201                while ((l & 1) == 0) {
202                l >>= 1;
203                }
204                if (!f(_op(s, _data[l]))) {
205                while (l < _size) {
206                    l <<= 1;
207                    if (f(_op(s, _data[l]))) {
208                    s = _op(s, _data[l]);
209                    l |= 1;
210                    }
211                }
212                return l - _size;
213                }
214                s = _op(s, _data[l]);
215                ++l;
216                if ((l & (-l)) == l) break;
217            }
218            return n;
219        }
220
221        template<typename F>  // F: function<bool (T)> f
222        int min_left(int r, F &&f) const {
223            assert(0 <= r && r <= n);
224            // assert(f(_e()));
225            if (r == 0) return 0;
226            r += _size;
227            T s = _e();
228            while (r > 0) {
229                --r;
230                while (r > 1 && (r & 1)) {
231                r >>= 1;
232                }
233                if (!f(_op(_data[r], s))) {
234                while (r < _size) {
235                    r = (r << 1) | 1;
236                    if (f(_op(_data[r], s))) {
237                    s = _op(_data[r], s);
238                    r ^= 1;
239                    }
240                }
241                return r + 1 - _size;
242                }
243                s = _op(_data[r], s);
244                if ((r & (-r)) == r) break;
245            }
246            return 0;
247        }
248
249        vector<T> tovector() const {
250            vector<T> res(n);
251            for (int i = 0; i < n; ++i) {
252                res[i] = get(i);
253            }
254            return res;
255        }
256
257        void print() const {
258            cout << '[';
259            for (int i = 0; i < n-1; ++i) {
260                cout << get(i) << ", ";
261            }
262            if (n > 0) cout << get(n-1);
263            cout << ']' << endl;
264        }
265    };
266}  // namespace titan23

272    /**
273     * @brief セグ木搭載HLD
274     * 
275     * @tparam T 
276     * @tparam (*op)(T, T) 
277     * @tparam (*e)() 
278     */
279    template <class T,
280              T (*op)(T, T),
281              T (*e)()>
282    class HLDSegmentTree {
283      private:
284        titan23::HLD hld;
285        titan23::SegmentTree<T, op, e> seg;
286
287      public:
288        HLDSegmentTree(const titan23::HLD &hld) {
289            this->hld = hld;
290            this->seg = titan23::SegmentTree<T, op, e>(hld.n);
291        }
292
293        HLDSegmentTree(const titan23::HLD &hld, constvector<T> &a) {
294            this->hld = hld;
295            this->seg = titan23::SegmentTree<T, op, e>(hld.build_list(a));
296        }
297
298        //! `u` から `v` へのパスの総積を返す / `O(logn)`
299        T path_prod(int u, int v) const {
300            T res = e();
301            while (hld.head[u] != hld.head[v]) {
302                if (hld.dep[hld.head[u]] < hld.dep[hld.head[v]]) {
303                    swap(u, v);
304                }
305                res = op(res, seg.prod(hld.nodein[hld.head[u]], hld.nodein[u]+1));
306                u = hld.par[hld.head[u]];
307            }
308            if (hld.dep[u] < hld.dep[v]) {
309                swap(u, v);
310            }
311            return op(res, seg.prod(hld.nodein[v], hld.nodein[u]+1));
312        }
313
314        //! 頂点 `k` の値を返す / `O(1)`
315        T get(const int k) const {
316            return seg.get(hld.nodein[k]);
317        }
318
319        //! 頂点 `k` の値を `v` に更新する / `O(logn)`
320        void set(const int k, const T v) {
321            seg.set(hld.nodein[k], v);
322        }
323
324        //! 頂点 `v` の部分木の総積を返す / `O(logn)`
325        T subtree_prod(const int v) const {
326            return seg.prod(hld.nodein[v], hld.nodeout[v]);
327        }
328    };
329}  // namespace titan23

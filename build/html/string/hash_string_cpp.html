  9  class HashStringBase {
 10    // ref: https://qiita.com/keymoon/items/11fac5627672a6d6a9f6
 11   public:
 12    using u64 = unsigned long long;
 13    static constexpr const u64 MASK31 = (1ULL << 31) - 1;
 14    static constexpr const u64 MASK30 = (1ULL << 30) - 1;
 15    static constexpr const u64 MOD = (1ULL << 61) - 1;
 16    static constexpr const u64 MASK61 = (1ULL << 61) - 1;
 17    int n;
 18    u64 base, invpow;
 19    vector<u64> powb, invb;
 20
 21    static u64 get_mul(u64 a, u64 b) {
 22      u64 au = a >> 31;
 23      u64 ad = a & MASK31;
 24      u64 bu = b >> 31;
 25      u64 bd = b & MASK31;
 26      u64 mid = ad * bu + au * bd;
 27      u64 midu = mid >> 30;
 28      u64 midd = mid & MASK30;
 29      return get_mod(au * bu * 2 + midu + (midd << 31) + ad * bd);
 30    }
 31
 32    static u64 get_mod(u64 x) {
 33      u64 xu = x >> 61;
 34      u64 xd = x & MASK61;
 35      u64 res = xu + xd;
 36      if (res >= MOD) res -= MOD;
 37      return res;
 38    }
 39
 40    u64 pow_mod(u64 a, u64 b, const u64 mod) {
 41      u64 res = 1ULL;
 42      while (b) {
 43        if (b & 1ULL) {
 44          res = get_mul(res, a);
 45        }
 46        a = get_mul(a, a);
 47        b >>= 1ULL;
 48      }
 49      return res;
 50    }
 51
 52    HashStringBase(int n = 0) : n(n) {
 53      std::mt19937 rand(std::random_device{}());
 54      // u64 base = std::uniform_int_distribution<>(37, 1000000000)(rand);
 55      u64 base = 414123123;
 56      this->base = base;
 57      this->invpow = pow_mod(base, HashStringBase::MOD-2, HashStringBase::MOD);
 58      powb.resize(n + 1, 1ULL);
 59      invb.resize(n + 1, 1ULL);
 60      for (int i = 1; i <= n; ++i) {
 61        powb[i] = get_mul(powb[i-1], base);
 62        invb[i] = get_mul(invb[i-1], this->invpow);
 63      }
 64    }
 65
 66    void extend(const int cap) {
 67      assert(cap >= 0);
 68      int pre_cap = powb.size();
 69      powb.resize(pre_cap + cap, 0);
 70      invb.resize(pre_cap + cap, 0);
 71      for (int i = pre_cap; i < pre_cap + cap; ++i) {
 72        powb[i] = get_mul(powb[i-1], base);
 73        invb[i] = get_mul(invb[i-1], invpow);
 74      }
 75    }
 76
 77    int get_cap() const {
 78      return powb.size();
 79    }
 80
 81    u64 unite(u64 h1, u64 h2, int k) {
 82      if (k >= get_cap()) extend(k - get_cap() + 1);
 83      return get_mod(get_mul(h1, powb[k]) + h2);
 84    }
 85  };

 87  class HashString {
 88   private:
 89    using u64 = unsigned long long;
 90    HashStringBase* hsb;
 91    int n;
 92    vector<u64> data, acc;
 93
 94   public:
 95    HashString() {}
 96    HashString(HashStringBase &hsb, string &s) : n(s.size()), hsb(&hsb) {
 97      data.resize(n);
 98      acc.resize(n, 0);
 99      if (n > this->hsb->get_cap()) this->hsb->extend(n - this->hsb->get_cap());
100      for (int i = 0 ; i < n; ++i) {
101        data[i] = this->hsb->get_mul(this->hsb->powb[n-i-1], s[i]-'a'+1);
102        acc[i+1] = this->hsb->get_mod(acc[i] + data[i]);
103      }
104    }
105
106    u64 get(int l, int r) const {
107      assert(0 <= l && l <= r && l <= n);
108      return hsb->get_mul(hsb->get_mod(4 * HashStringBase::MOD + acc[r] - acc[l]), hsb->invb[n - r]);
109    }
110
111    vector<int> get_lcp() const {
112      vector<int> a(n, 0);
113      for (int i = 0; i < n; ++i) {
114        int ok = 0, ng = n - i + 1;
115        while (ng - ok > 1) {
116          int mid = (ok + ng) / 2;
117          (get(0, mid) == get(i, i + mid) ? ok : ng) = mid;
118        }
119        a[i] = ok;
120      }
121      return a;
122    }
123  };

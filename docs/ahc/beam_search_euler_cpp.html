 20      private:
 21        unsigned int _x, _y, _z, _w;

 23        unsigned int _xor128() {
 24            const unsigned int t = _x ^ (_x << 11);
 25            _x = _y;
 26            _y = _z;
 27            _z = _w;
 28            _w = (_w ^ (_w >> 19)) ^ (t ^ (t >> 8));
 29            return _w;
 30        }

 33        Random() : _x(123456789),
 34                   _y(362436069),
 35                   _z(521288629),
 36                   _w(88675123) {}

 38        //! `[0, 1]` の乱数を返す(実数)
 39        double random() { return (double)(_xor128()) / 0xFFFFFFFF; }

 41        //! `[0, end]` の乱数を返す(整数)
 42        int randint(const int end) {

 66        unsigned long long rand_u64() {
 67            return ((unsigned long long)_xor128() << 32) | _xor128();

 70        //! `[0, end)` の異なる乱数を2つ返す
 71        pair<int, int> rand_pair(const int end) {
 72            assert(end >= 2);
 73            int u = randrange(end);
 74            int v = u + randrange(1, end);
 75            if (v >= end) v -= end;
 76            if (u > v) swap(u, v);

120            vector<T> r(k);
121            unordered_set<int> seen;
122            for (int i = 0; i < k; ++i) {
123                int x = randrange(a.size());
124                while (seen.find(x) != seen.end()) x = randrange(a.size());

136        template <typename T>
137        T choice(const vector<T> &a, const vector<int> &w, bool normal) {
138            assert(normal == false);
139            assert(a.size() == w.size());
140            double sum = 0.0;
141            for (const int &x: w) sum += x;
142            assert(sum > 0);
143            vector<double> x(w.size());
144            for (int i = 0; i < x.size(); ++i) {
145                x[i] = (double)w[i] / sum;
146                if (i-1 >= 0) x[i] += x[i-1];
147            }
148            return choice(a, x);
149        }

151        template <typename T>
152        T choice(const vector<T> &a, const vector<double> &w) {
153            double i = random();
154            int l = -1, r = a.size()-1;
155            while (r - l > 1) {
156                int mid = (l + r) / 2;
157                if (w[mid] <= i) l = mid;
158                else r = mid;
159            }
160            return a[r];

176    template<typename T>
177    class StatePool {
178      private:
179        vector<T*> pool;
180        stack<int> unused_indx;
181
182      public:
183        StatePool() {}
184        StatePool(const int n) { init(n); }
185
186        //! clear
187        void clear() {
188            while (!unused_indx.empty()) unused_indx.pop();
189            for (int i = (int)pool.size()-1; i >= 0; --i) {
190                unused_indx.emplace(i);
191            }
192        }
193
194        //! n要素確保する。
195        void init(const int n) {
196            for (int i = 0; i < n; ++i) {
197                T* state = new T;
198                pool.emplace_back(state);
199                unused_indx.emplace(i);
200            }
201        }
202
203        //! id に対応する T のポインタを返す。
204        T* get(int id) const {
205            assert(0 <= id && id < pool.size());
206            return pool[id];
207        }
208
209        //! idに対応するTを仮想的に削除する。
210        void del(int id) {
211            assert(0 <= id && id < pool.size());
212            unused_indx.emplace(id);
213        }
214
215        //! T を作成し、それに対応する id を返す。
216        int gen() {
217            int state_id;
218            if (unused_indx.empty()) {
219                T* state = new T;
220                state_id = pool.size();
221                pool.emplace_back(state);
222            } else {
223                state_id = unused_indx.top();
224                unused_indx.pop();
225            }
226            return state_id;
227        }
228
229        //! id に対応するTをコピーし、コピー先のidを返す。
230        //! T のコピーメソッドを呼び出す。
231        int copy(const int id) {
232            int new_id = gen();
233            pool[id]->copy(pool[new_id]);
234            return new_id;
235        }
236
237        //! 内部サイズを呼び出す
238        int get_size() const {
239            return pool.size();
240        }
241    };

290    class HashSet {
291      private:
292        using u64 = unsigned long long;
293        static constexpr const u64 K = 0x517cc1b727220a95;
294        static constexpr const int M = 2;
295        vector<u64> exist;
296        vector<u64> keys;
297        int msk, xor_;
298        int size;
299
300        int hash(const u64 &key) const {
301            return (((((key>>32)&msk) ^ (key&msk) ^ xor_)) * (HashSet::K & msk)) & msk;
302        }
303
304        pair<int, bool> get_pos(const u64 &key) const {
305            int h = hash(key);
306            while (true) {
307                if (!(exist[h>>6]>>(h&63)&1)) return {h, false};
308                if (keys[h] == key) return {h, true};
309                h = (h + 1) & msk;
310            }
311        }
312
313        int bit_length(const int x) const {
314            if (x == 0) return 0;
315            return 32 - __builtin_clz(x);
316        }
317
318        void rebuild() {
319            vector<u64> old_exist = exist;
320            vector<u64> old_keys = keys;
321            exist.resize(HashSet::M*old_exist.size()+1);
322            fill(exist.begin(), exist.end(), 0);
323            keys.resize(HashSet::M*old_keys.size());
324            size = 0;
325            msk = (1<<bit_length(keys.size()-1))-1;
326            random_device rd;
327            mt19937 gen(rd());
328            uniform_int_distribution<int> dis(0, msk);
329            xor_ = dis(gen);
330            for (int i = 0; i < (int)old_keys.size(); ++i) {
331                if (old_exist[i>>6]>>(i&63)&1) {
332                    insert(old_keys[i]);
333                }
334            }
335        }
336
337      public:
338        HashSet() : exist(1, 0), keys(1), msk(0), xor_(0), size(0) {}
339
340        HashSet(const int n) {
341            int s = 1<<bit_length(n);
342            s *= HashSet::M;
343            assert(s > 0);
344            exist.resize((s>>6)+1, 0);
345            keys.resize(s);
346            msk = (1<<bit_length(keys.size()-1))-1;
347            random_device rd;
348            mt19937 gen(rd());
349            uniform_int_distribution<int> dis(0, msk);
350            xor_ = dis(gen);
351            size = 0;
352        }
353
354        bool contains(const u64 key) const {
355            return get_pos(key).second;
356        }
357
358        void insert(const u64 key) {
359            const auto [pos, is_exist] = get_pos(key);
360            if (!is_exist) {
361                exist[pos>>6] |= 1ull<<(pos&63);
362                keys[pos] = key;
363                ++size;
364                if (HashSet::M*size > keys.size()) {
365                    rebuild();
366                }
367            }
368        }
369
370        //! keyがすでにあればtrue, なければ挿入してfalse / `O(1)`
371        bool contains_insert(const u64 key) {
372            const auto [pos, is_exist] = get_pos(key);
373            if (!is_exist) {
374                exist[pos>>6] |= 1ull<<(pos&63);
375                keys[pos] = key;
376                ++size;
377                if (HashSet::M*size > keys.size()) {
378                    rebuild();
379                }
380                return false;
381            }
382            return true;
383        }
384
385        //! 全ての要素を削除する / `O(n/w)`
386        void clear() {
387            this->size = 0;
388            fill(exist.begin(), exist.end(), 0);
389        }
390
391        int len() const {
392            return size;
393        }
394    };

559class BeamSearchWithTree {
560    // ref: https://eijirou-kyopro.hatenablog.com/entry/2024/02/01/115639
561
562  private:
563    static constexpr const int PRE_ORDER = -1;
564    static constexpr const int POST_ORDER = -2;
565
566    titan23::HashSet seen;
567    using ActionIDType = int;
568    ActionIDType ActionID;
569    vector<Action> result;
570
571    // ビームサーチの過程を表す木
572    // <dir or id, action, action_id>
573    // dir or id := 葉のとき、leaf_id
574    //              そうでないとき、行きがけなら-1、帰りがけなら-2
575    vector<tuple<int, Action, ActionIDType>> tree;
576
577    // 次のビーム候補を保持する配列
578    vector<tuple<int, ScoreType, Action, ActionIDType>> next_beam; // <par, score, action, action_id>
579
580    vector<vector<int>> next_beam_data;
581
582    void get_next_beam(State* state, const int turn) {
583        next_beam.clear();
584        next_beam.reserve(tree.size() * 4); // TODO
585        seen.clear();
586
587        if (turn == 0) {
588            vector<Action> actions = state->get_actions();
589            for (Action &action : actions) {
590                auto [score, hash] = state->try_op(action);
591                if (seen.contains_insert(hash)) continue;
592                next_beam.emplace_back(PRE_ORDER, score, action, ActionID);
593                ActionID++;
594            }
595            return;
596        }
597
598        int leaf_id = 0;
599        for (int i = 0; i < tree.size(); ++i) {
600            auto [dir_or_leaf_id, action, _] = tree[i];
601            if (dir_or_leaf_id >= 0) {
602                state->apply_op(action);
603                vector<Action> actions = state->get_actions();
604                std::get<0>(tree[i]) = leaf_id;
605                for (Action &action : actions) {
606                    auto [score, hash] = state->try_op(action);
607                    if (seen.contains_insert(hash)) continue;
608                    next_beam.emplace_back(leaf_id, score, action, ActionID);
609                    ActionID++;
610                }
611                ++leaf_id;
612                state->rollback(action);
613            } else if (dir_or_leaf_id == PRE_ORDER) {
614                state->apply_op(action);
615            } else {
616                state->rollback(action);
617            }
618        }
619    }
620
621    //! 不要なNodeを削除し、木を更新する
622    int update_tree(State* state, const int turn) {
623        vector<tuple<int, Action, ActionIDType>> new_tree;
624        new_tree.reserve(tree.size());
625        if (turn == 0) {
626            for (auto &[par, _, new_action, action_id] : next_beam) {
627                assert(par == -1);
628                new_tree.emplace_back(0, new_action, action_id);
629            }
630            swap(tree, new_tree);
631            return 0;
632        }
633
634        int i = 0;
635        int apply_only_turn = 0;
636        while (true) {
637            const auto &[dir_or_leaf_id, action, action_id] = tree[i];
638            // 行きがけかつ帰りがけのaction_idが一致しているなら、一本道なので行くだけ
639            if (dir_or_leaf_id == PRE_ORDER && action_id == std::get<2>(tree.back())) {
640                ++i;
641                result.emplace_back(action);
642                state->apply_op(action);
643                tree.pop_back();
644                apply_only_turn++;
645            } else {
646                break;
647            }
648        }
649
650        for (; i < tree.size(); ++i) {
651            const auto &[dir_or_leaf_id, action, action_id] = tree[i];
652            if (dir_or_leaf_id >= 0) {
653                if (next_beam_data[dir_or_leaf_id].empty()) continue;
654                new_tree.emplace_back(PRE_ORDER, action, action_id);
655                for (const int beam_idx : next_beam_data[dir_or_leaf_id]) {
656                    auto &[_, __, new_action, new_action_id] = next_beam[beam_idx];
657                    new_tree.emplace_back(dir_or_leaf_id, new_action, new_action_id);
658                }
659                new_tree.emplace_back(POST_ORDER, action, action_id);
660                next_beam_data[dir_or_leaf_id].clear();
661            } else if (dir_or_leaf_id == PRE_ORDER) {
662                new_tree.emplace_back(PRE_ORDER, action, action_id);
663            } else {
664                int pre_dir = std::get<0>(new_tree.back());
665                if (pre_dir == PRE_ORDER) {
666                    new_tree.pop_back(); // 一つ前が行きがけなら、削除して追加しない
667                } else {
668                    new_tree.emplace_back(POST_ORDER, action, action_id);
669                }
670            }
671        }
672        swap(tree, new_tree);
673        return apply_only_turn;
674    }
675
676    void get_result() {
677        int best_id = -1;
678        ScoreType best_score = 0;
679        for (auto [par, score, _, __] : next_beam) {
680            if (best_id == -1 || score < best_score) {
681                best_score = score;
682                best_id = par;
683            }
684        }
685        assert(best_id != -1);
686        for (const auto &[dir_or_leaf_id, action, _] : tree) {
687            if (dir_or_leaf_id >= 0) {
688                if (best_id == dir_or_leaf_id) {
689                    result.emplace_back(action);
690                    return;
691                }
692            } else if (dir_or_leaf_id == PRE_ORDER) {
693                result.emplace_back(action);
694            } else {
695                result.pop_back();
696            }
697        }
698        cerr << PRINT_RED << "Error: 解が見つかりませんでした" << PRINT_NONE << endl;
699        assert(false);
700    }
701
702  public:
703
704    /**
705     * @brief ビームサーチをする
706     *
707     * @param param ターン数、ビーム幅を指定するパラメータ構造体
708     * @param verbose ログ出力するかどうか
709     * @return vector<Action>
710     */
711    vector<Action> search(const BeamParam &param, const bool verbose = false) {
712        if (verbose) cerr << PRINT_GREEN << "Info: start search()" << PRINT_NONE << endl;
713
714        ActionID = 0;
715        State* state = new State;
716        state->init();
717
718        this->seen = titan23::HashSet(param.BEAM_WIDTH * 4); // TODO
719
720        int now_turn = 0;
721        for (int turn = 0; turn < param.MAX_TURN; ++turn) {
722            if (verbose) cerr << "Info: # turn : " << turn+1 << endl;
723
724            // 次のビーム候補を求める
725            get_next_beam(state, turn-now_turn);
726
727            if (next_beam.empty()) {
728                cerr << PRINT_RED << "Error: \t次の候補が見つかりませんでした" << PRINT_NONE << endl;
729                assert(!next_beam.empty());
730            }
731
732            // ビームを絞る
733            int beam_width = min(param.BEAM_WIDTH, (int)next_beam.size());
734            assert(beam_width <= param.BEAM_WIDTH);
735
736            nth_element(next_beam.begin(), next_beam.begin() + beam_width, next_beam.end(), [&] (const tuple<int, ScoreType, Action, ActionIDType> &left, const tuple<int, ScoreType, Action, ActionIDType> &right) {
737                return std::get<1>(left) < std::get<1>(right);
738            });
739
740            tuple<int, ScoreType, Action, ActionIDType> bests = *min_element(next_beam.begin(), next_beam.begin() + beam_width, [&] (const tuple<int, ScoreType, Action, ActionIDType> &left, const tuple<int, ScoreType, Action, ActionIDType> &right) {
741                return std::get<1>(left) < std::get<1>(right);
742            });
743            if (verbose) cerr << "Info: \tbest_score = " << std::get<1>(bests) << endl;
744            if (std::get<1>(bests) == 0) { // TODO 終了条件
745                cerr << PRINT_GREEN << "Info: find valid solution." << PRINT_NONE << endl;
746                get_result();
747                result.emplace_back(std::get<2>(bests));
748                return result;
749            }
750
751            // 探索木の更新
752            if (turn != 0) {
753                if (next_beam_data.size() < next_beam.size()) {
754                    next_beam_data.resize(next_beam.size());
755                }
756                for (int i = 0; i < beam_width; ++i) {
757                    auto &[par, _, new_action, new_action_id] = next_beam[i];
758                    next_beam_data[par].emplace_back(i);
759                }
760            }
761            int apply_only_turn = update_tree(state, turn);
762            now_turn += apply_only_turn;
763        }
764
765        // 答えを復元する
766        if (verbose) cerr << PRINT_GREEN << "Info: MAX_TURN finished." << PRINT_NONE << endl;
767        get_result();
768        return result;
769    }
770};
771} // namespace beam_search

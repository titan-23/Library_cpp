  9  struct BinaryTrieSet {
 10    vector<unsigned int> left, right, par, size;
 11    unsigned int _end, _root, _bit;
 12    T _lim, _xor_val;
 13
 14    BinaryTrieSet(unsigned int bit) {
 15      _end = 2;
 16      _bit = bit;
 17      _root = 1;
 18      _lim = 1ll << bit;
 19      _xor_val = 0;
 20      left.resize(2);
 21      right.resize(2);
 22      par.resize(2);
 23      size.resize(2);
 24    }
 25
 26    void reserve(const int n) {
 27      left.reserve(n);
 28      right.reserve(n);
 29      par.reserve(n);
 30      size.reserve(n);
 31    }
 32
 33    unsigned int _make_node() {
 34      if (_end >= (int)left.size()) {
 35        left.emplace_back(0);
 36        right.emplace_back(0);
 37        par.emplace_back(0);
 38        size.emplace_back(0);
 39      }
 40      return _end++;
 41    }
 42
 43    unsigned int _find(T key) const {
 44      key ^= _xor_val;
 45      unsigned int node = _root;
 46      for (int i = _bit-1; i >= 0; --i) {
 47        if ((key >> i) & 1) {
 48          if (!right[node]) return -1;
 49          node = right[node];
 50        } else {
 51          if (!left[node]) return -1;
 52          node = left[node];
 53        }
 54      }
 55      return node;
 56    }
 57
 58    bool add(T key) {
 59      key ^= _xor_val;
 60      unsigned int node = _root;
 61      for (int i = _bit-1; i >= 0; --i) {
 62        if ((key >> i) & 1) {
 63          if (!right[node]) {
 64            unsigned int new_node = _make_node();
 65            right[node] = new_node;
 66            par[right[node]] = node;
 67          }
 68          node = right[node];
 69        } else {
 70          if (!left[node]) {
 71            unsigned int new_node = _make_node();
 72            left[node] = new_node;
 73            par[left[node]] = node;
 74          }
 75          node = left[node];
 76        }
 77      }
 78      if (size[node]) return false;
 79      size[node] = 1;
 80      for (int i = 0; i < _bit; ++i) {
 81        node = par[node];
 82        size[node] += 1;
 83      }
 84      return true;
 85    }
 86
 87    bool contains(const T key) const {
 88      return _find(key) != -1;
 89    }
 90
 91    void _discard(unsigned int node) {
 92      for (int i = 0; i < _bit; ++i) {
 93        size[node] -= 1;
 94        if (left[par[node]] == node) {
 95          node = par[node];
 96          left[node] = 0;
 97          if (right[node]) break;
 98        } else {
 99          node = par[node];
100          right[node] = 0;
101          if (left[node]) break;
102        }
103      }
104      while (node) {
105        size[node] -= 1;
106        node = par[node];
107      }
108    }
109
110    bool discard(T key) {
111      unsigned int node = _find(key);
112      if (node == -1) return false;
113      _discard(node);
114      return true;
115    }
116
117    T pop(int k) {
118      if (k < 0) k += len();
119      int node = _root;
120      T res = 0;
121      for (int i = _bit-1; i >= 0; --i) {
122        if ((_xor_val >> i) & 1) swap(left, right);
123        unsigned int t = size[left[node]];
124        res <<= 1;
125        if (t <= k) {
126          k -= t;
127          res |= 1;
128          node = right[node];
129        } else {
130          node = left[node];
131        }
132        if ((_xor_val >> i) & 1) swap(left, right);
133      }
134      _discard(node);
135      return res ^ _xor_val;
136    }
137
138    T pop_min() {
139      return pop(0);
140    }
141
142    T pop_max() {
143      return pop(-1);
144    }
145
146    void all_xor(T x) {
147      _xor_val ^= x;
148    }
149
150    T get_min() const {
151      T key = _xor_val;
152      T ans = 0;
153      unsigned int node = _root;
154      for (int i = _bit-1; i >= 0; --i) {
155        ans <<= 1;
156        if ((key >> i) & 1) {
157          if (right[node]) {
158            node = right[node];
159            ans |= 1;
160          } else {
161            node = left[node];
162          }
163        } else {
164          if (left[node]) {
165            node = left[node];
166          } else {
167            node = right[node];
168            ans |= 1;
169          }
170        }
171      }
172      return ans ^ _xor_val;
173    }
174
175    T get_max() const {
176      T key = _xor_val;
177      T ans = 0;
178      unsigned int node = _root;
179      for (int i = _bit-1; i >= 0; --i) {
180        ans <<= 1;
181        if ((key >> i) & 1) {
182          if (left[node]) {
183            node = left[node];
184          } else {
185            node = right[node];
186            ans |= 1;
187          }
188        } else {
189          if (right[node]) {
190            ans |= 1;
191            node = right[node];
192          } else {
193            node = left[node];
194          }
195        }
196      }
197      return ans ^ _xor_val;
198    }
199
200    int index(T key) const {
201      int k = 0;
202      unsigned int node = _root;
203      key ^= _xor_val;
204      for (int i = _bit-1; i >= 0; --i) {
205        if ((key >> i) & 1) {
206          k += size[left[node]];
207          node = right[node];
208        } else {
209          node = left[node];
210        }
211        if (!node) break;
212      }
213      return k;
214    }
215
216    int index_right(T key) const {
217      int k = 0;
218      unsigned int node = _root;
219      key ^= _xor_val;
220      for (int i = _bit-1; i >= 0; --i) {
221        if ((key >> i) & 1) {
222          k += size[left[node]];
223          node = right[node];
224        } else {
225          node = left[node];
226        }
227        if (!node) break;
228      }
229      if (node) k += 1;
230      return k;
231    }
232
233    T kth_elm(int k) {
234      if (k < 0) k += len();
235      unsigned int node = _root;
236      T res = 0;
237      for (int i = _bit-1; i >= 0; --i) {
238        if ((_xor_val >> i) & 1) swap(left, right);
239        unsigned int t = size[left[node]];
240        res <<= 1;
241        if (t <= k) {
242          k -= t;
243          res |= 1;
244          node = right[node];
245        } else {
246          node = left[node];
247        }
248        if ((_xor_val >> i) & 1) swap(left, right);
249      }
250      return res;
251    }
252
253    T gt(T key) {
254      int i = index_right(key);
255      return (i >= size[_root]? (-1) : kth_elm(i));
256    }
257
258    T lt(T key) {
259      int i = index(key) - 1;
260      return (i < 0? -1 : kth_elm(i));
261    }
262
263    T ge(T key) {
264      if (key == 0) return (len()? get_min() : -1);
265      int i = index_right(key - 1);
266      return (i >= size[_root]? -1 : kth_elm(i));
267    }
268
269    T le(T key) {
270      int i = index(key + 1) - 1;
271      return (i < 0? -1 : kth_elm(i));
272    }
273
274    vector<T> tovector() {
275      vector<T> a;
276      if (!len()) return a;
277      a.reserve(len());
278      for (int i = 0; i < len(); ++i) {
279        T e = kth_elm(i);
280        a.emplace_back(e);
281      }
282      return a;
283    }
284
285    bool empty() const {
286      return size[_root] == 0;
287    }
288
289    void print() {
290      cout << "{";
291      vector<T> a = tovector();
292      for (int i = 0; i < len()-1; ++i) {
293        cout << a[i] << ", ";
294      }
295      if (!a.empty()) {
296        cout << a.back();
297      }
298      cout << "}" << endl;
299    }
300
301    int len() const {
302      return size[_root];
303    }
304  };

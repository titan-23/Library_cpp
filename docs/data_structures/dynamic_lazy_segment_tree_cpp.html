  6// DynamicLazySegmentTree
  7namespace titan23 {
  8
  9    /**
 10     * @brief `[0, u)` の列を管理する、必要なところだけ作る遅延セグ木
 11     *
 12     * - apply / set : `O(logu)` time, `O(logu)` space
 13     *
 14     * - prod  / get : `O(logu)` time, `O(1)` space
 15     *
 16     * @tparam IndexType 添え字を表すインデックス long long を推奨 和がオーバーフローしないことが条件かな
 17     * @tparam T モノイドの型
 18     * @tparam (*op)(T, T) モノイドの2項演算
 19     * @tparam (*e)() モノイドの単位元
 20     * @tparam F 作用素の型
 21     * @tparam (*mapping)(F, T) 遅延セグ木のアレ
 22     * @tparam (*composition)(F, F) 遅延セグ木のアレ
 23     * @tparam (*id)() 遅延セグ木のアレ
 24     * @tparam (*pow)(T, IndexType) T=op(T,T)をk回繰り返した結果を返す関数
 25     */
 26    template <class IndexType,
 27            class T,
 28            T (*op)(T, T),
 29            T (*e)(),
 30            class F,
 31            T (*mapping)(F, T),
 32            F (*composition)(F, F),
 33            F (*id)(),
 34            T (*pow)(T, IndexType)>
 35    class DynamicLazySegmentTree {
 36      private:
 37        static int bit_length(long long x) {
 38            return x == 0 ? 0 : 64 - __builtin_clzll(x);
 39        }
 40
 41        struct Node;
 42        using NodePtr = Node*;
 43        // using NodePtr = shared_ptr<Node>;
 44        NodePtr root;
 45        IndexType u;
 46
 47        struct Node {
 48            NodePtr left, right;
 49            IndexType l, r;
 50            T key, data; // 子がないとき、data=pow(key, r-l)が成り立つ
 51            F lazy;
 52
 53            Node() {}
 54
 55            Node(IndexType l, IndexType r, T key) :
 56                    left(nullptr), right(nullptr),
 57                    l(l), r(r),
 58                    key(key), data(pow(key, r-l)) {
 59                assert(l < r);
 60            }
 61
 62            bool is_leaf() const {
 63                return this->r - this->l == 1;
 64            }
 65
 66            void apply(const F f) {
 67                if (f == id()) return;
 68                this->key = mapping(f, this->key);
 69                this->data = mapping(f, this->data);
 70                if (!is_leaf()) this->lazy = composition(f, this->lazy);
 71            }
 72
 73            void propagate() {
 74                if (is_leaf()) return;
 75                if (this->left) {
 76                    this->left->apply(this->lazy);
 77                } else {
 78                    this->left = new Node(l, mid(), this->key);
 79                }
 80                if (this->right) {
 81                    this->right->apply(this->lazy);
 82                } else {
 83                    this->right = new Node(mid(), r, this->key);
 84                }
 85                this->lazy = id();
 86            }
 87
 88            void update() {
 89                if (this->left) {
 90                    this->data = this->left->data;
 91                } else {
 92                    this->data = e();
 93                }
 94                if (this->right) {
 95                    this->data = op(this->data, this->right->data);
 96                }
 97            }
 98
 99            IndexType mid() const {
100                return (this->l + this->r) / 2;
101            }
102        };
103
104      private:
105        T inner_prod(NodePtr node, IndexType l, IndexType r) {
106            if (!node || l >= r || r <= node->l || node->r <= l) return e();
107            if (l <= node->l && node->r <= r) return node->data;
108            if ((!node->left) && (!node->right)) return pow(node->key, r - l);
109            node->propagate();
110            return op(
111                inner_prod(node->left, l, min(r, node->mid())),
112                inner_prod(node->right, max(l, node->mid()), r)
113            );
114        }
115
116        void inner_apply(NodePtr node, IndexType l, IndexType r, F f) {
117            if (!node || l >= r || r <= node->l || node->r <= l) return;
118            if (l <= node->l && node->r <= r) {
119                node->apply(f);
120                return;
121            }
122            node->propagate();
123            inner_apply(node->left, l, min(r, node->mid()), f);
124            inner_apply(node->right, max(l, node->mid()), r, f);
125            node->update();
126        }
127
128        void inner_set(NodePtr node, IndexType k, T val) {
129            assert(node);
130            if (node->is_leaf()) {
131                assert(node->l == k && node->r == k+1);
132                node->key = val;
133                node->data = val;
134                return;
135            }
136            node->propagate();
137            assert(node->left && node->right);
138            if (k < node->mid()) {
139                inner_set(node->left, k, val);
140            } else {
141                inner_set(node->right, k, val);
142            }
143            node->update();
144        }
145
146        T inner_get(NodePtr node, IndexType k) {
147            assert(node);
148            while (true) {
149                if (node->is_leaf()) {
150                    assert(node->l == k && node->r == k+1);
151                    return node->key;
152                }
153                if (k < node->mid()) {
154                    if (!node->left) return node->key;
155                    node->propagate();
156                    node = node->left;
157                } else {
158                    if (!node->right) return node->key;
159                    node->propagate();
160                    node = node->right;
161                }
162            }
163        }
164
165      public:
166
167        DynamicLazySegmentTree() : root(nullptr), u(0) {}
168
169        //! 初期値 `e()` , `[0, u)` の区間を管理する `DynamicLazySegmentTree` を作成する
170        DynamicLazySegmentTree(const IndexType u_) {
171            assert(u_ > 0);
172            this->u = 1ll << bit_length(u_);
173            this->root = new Node(0, this->u, e());
174        }
175
176        //! 初期値 `init` , `[0, u)` の区間を管理する `DynamicLazySegmentTree` を作成する
177        DynamicLazySegmentTree(const IndexType u_, const T init) {
178            assert(u_ > 0);
179            this->u = 1ll << bit_length(u_);
180            this->root = new Node(0, this->u, init);
181        }
182
183        //! `[l, r)` の集約値を返す / `O(logu)` time, `O(1)` space
184        T prod(IndexType l, IndexType r) {
185            assert(0 <= l && l <= r && r <= u);
186            return inner_prod(this->root, l, r);
187        }
188
189        //! `[l, r)` に `f` を作用させる / `O(logu)` time, `O(logu)` space
190        void apply(IndexType l, IndexType r, F f) {
191            assert(0 <= l && l <= r && r <= u);
192            inner_apply(this->root, l, r, f);
193        }
194
195        //! `k` 番目の値を取得する / `O(logu)` time, `O(1)` space
196        T get(IndexType k) {
197            return inner_get(this->root, k);
198        }
199
200        //! `k` 番目の値を `val` に更新する / `O(logu)` time, `O(logu)` space
201        void set(IndexType k, T val) {
202            assert(0 <= k && k < u);
203            inner_set(this->root, k, val);
204        }
205
206        //! 適当に表示する
207        void print() {
208            for (IndexType i = 0; i < u; ++id) {
209                cout << get(i) << ", ";
210            }
211            cout << endl;
212        }
213    };
214}  // namespace titan23

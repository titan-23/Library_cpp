 24    static constexpr double BETA = (1 - 2 * ALPHA) / (1 - ALPHA);

 26    class Node {
 27
 28     public:
 29      T key, data;
 30      NodePtr left;
 31      NodePtr right;
 32      F lazy;
 33      int rev, size;
 34
 35      Node(T key, F lazy) : key(key), data(key), left(nullptr), right(nullptr), lazy(lazy), rev(0), size(1) {}
 36
 37      double balance() const {
 38        return ((left ? left->size : 0) + 1.0) / (size + 1.0);
 39      }
 40
 41      void _update() {
 42        size = 1;
 43        data = key;
 44        if (left) {
 45          size += left->size;
 46          data = op(left->data, key);
 47        }
 48        if (right) {
 49          size += right->size;
 50          data = op(data, right->data);
 51        }
 52      }
 53
 54      void _propagate() {
 55        if (rev) {
 56          swap(left, right);
 57          if (left) left->rev ^= 1;
 58          if (right) right->rev ^= 1;
 59          rev = 0;
 60        }
 61        if (lazy != id()) {
 62          if (left) {
 63            left->key = mapping(lazy, left->key);
 64            left->data = mapping(lazy, left->data);
 65            left->lazy = composition(lazy, left->lazy);
 66          }
 67          if (right) {
 68            right->key = mapping(lazy, right->key);
 69            right->data = mapping(lazy, right->data);
 70            right->lazy = composition(lazy, right->lazy);
 71          }
 72          lazy = id();
 73        }
 74      }
 75
 76      void print() const {
 77        cout << "this : key=" << key << ", size=" << size << endl;
 78        if (left)  cout << "to-left" << endl;
 79        if (right) cout << "to-right" << endl;
 80        cout << endl;
 81        if (left)  left->print();
 82        if (right) right->print();
 83      }

 86    void _build(vector<T> const &a) {
 87      auto build = [&] (auto &&build, int l, int r) -> NodePtr {
 88        int mid = (l + r) >> 1;
 89        NodePtr node = new Node(a[mid], id());
 90        if (l != mid) node->left = build(build, l, mid);
 91        if (mid+1 != r) node->right = build(build, mid+1, r);
 92        node->_update();
 93        return node;
 94      };
 95      root = build(build, 0, (int)a.size());
 96    }

 98    NodePtr _rotate_right(NodePtr node) {
 99      NodePtr u = node->left;
100      node->left = u->right;
101      u->right = node;
102      node->_update();
103      u->_update();
104      return u;
105    }

107    NodePtr _rotate_left(NodePtr node) {
108      NodePtr u = node->right;
109      node->right = u->left;
110      u->left = node;
111      node->_update();
112      u->_update();
113      return u;
114    }

116    NodePtr _balance_left(NodePtr node) {
117      NodePtr u = node->right;
118      u->_propagate();
119      if (u->balance() >= BETA) {
120        u->left->_propagate();
121        node->right = _rotate_right(u);
122      }
123      return _rotate_left(node);
124    }
125
126    NodePtr _balance_right(NodePtr node) {
127      NodePtr u = node->left;
128      u->_propagate();
129      if (u->balance() <= 1.0 - BETA) {
130        u->right->_propagate();
131        node->left = _rotate_left(u);
132      }
133      return _rotate_right(node);
134    }

136    NodePtr _merge_with_root(NodePtr l, NodePtr root, NodePtr r) {
137      int ls = l? l->size: 0;
138      int rs = r? r->size: 0;
139      double diff = (double)(ls+1.0) / (ls+rs+2.0);
140      if (diff > 1.0-ALPHA) {
141        l->_propagate();
142        l->right = _merge_with_root(l->right, root, r);
143        l->_update();
144        if (!(ALPHA <= l->balance() && l->balance() <= 1.0-ALPHA)) {
145          return _balance_left(l);
146        }
147        return l;
148      }
149      if (diff < ALPHA) {
150        r->_propagate();
151        r->left = _merge_with_root(l, root, r->left);
152        r->_update();
153        if (!(ALPHA <= r->balance() && r->balance() <= 1.0-ALPHA)) {
154          return _balance_right(r);
155        }
156        return r;
157      }
158      root->left = l;
159      root->right = r;
160      root->_update();
161      return root;
162    }

164    pair<NodePtr, NodePtr> _pop_right(NodePtr node) {
165      vector<NodePtr> path;
166      node->_propagate();
167      NodePtr mx = node;
168      while (node->right) {
169        path.emplace_back(node);
170        node = node->right;
171        node->_propagate();
172        mx = node;
173      }
174      path.emplace_back(node->left? node->left: nullptr);
175      while ((int)path.size() > 1) {
176        node = path.back();
177        path.pop_back();
178        if (!node) {
179          path.back()->right = nullptr;
180          path.back()->_update();
181          continue;
182        }
183        double b = node->balance();
184        if (ALPHA <= b && b <= 1.0-ALPHA) {
185          path.back()->right = node;
186        } else if (b > 1.0-ALPHA) {
187          path.back()->right = _balance_right(node);
188        } else {
189          path.back()->right = _balance_left(node);
190        }
191        path.back()->_update();
192      }
193      if (path[0]) {
194        double b = path[0]->balance();
195        if (b > 1.0-ALPHA) {
196          path[0] = _balance_right(path[0]);
197        } else if (b < ALPHA) {
198          path[0] = _balance_left(path[0]);
199        }
200      }
201      mx->left = nullptr;
202      mx->_update();
203      return {path[0], mx};
204    }

206    NodePtr _merge_node(NodePtr l, NodePtr r) {
207      if ((!l) && (!r)) {return nullptr;}
208      if (!l) {return r;}
209      if (!r) {return l;}
210      auto [l_, root_] = _pop_right(l);
211      return _merge_with_root(l_, root_, r);
212    }

214    pair<NodePtr, NodePtr> _split_node(NodePtr node, int k) {
215      if (!node) return {nullptr, nullptr};
216      node->_propagate();
217      int tmp = node->left? k-node->left->size: k;
218      NodePtr nl = node->left;
219      NodePtr nr = node->right;
220      if (tmp == 0) {
221        return {nl, _merge_with_root(nullptr, node, nr)};
222      } else if (tmp < 0) {
223        auto [l, r] = _split_node(nl, k);
224        return {l, _merge_with_root(r, node, nr)};
225      } else {
226        auto [l, r] = _split_node(nr, tmp-1);
227        return {_merge_with_root(node->left, node, l), r};
228      }
229    }

255    void apply(const int l, const int r, const F f) {
256      if (l >= r) return;
257      auto [s_, t] = _split_node(root, r);
258      auto [u, s] = _split_node(s_, l);
259      s->key = mapping(f, s->key);
260      s->data = mapping(f, s->data);
261      s->lazy = composition(f, s->lazy);
262      this->root = _merge_node(_merge_node(u, s), t);
263    }

265    T prod(const int l, const int r) {
266      if (l >= r) return e();
267      auto [s_, t] = _split_node(root, r);
268      auto [u, s] = _split_node(s_, l);
269      T res = s->data;
270      this->root = _merge_node(_merge_node(u, s), t);
271      return res;
272    }

274    void insert(int k, const T key) {
275      auto [s, t] = _split_node(root, k);
276      NodePtr new_node = new Node(key, id());
277      this->root = _merge_with_root(s, new_node, t);

280    T pop(int k) {
281      if (k < 0) k += len();
282      assert(0 <= k && k < len());
283      vector<NodePtr> path;
284      vector<short> path_d;
285      NodePtr node = this->root;
286      T res;
287      int d = 0;
288      while (1) {
289        node->_propagate();
290        int t = node->left? node->left->size: 0;
291        if (t == k) {
292          res = node->key;
293          break;
294        }
295        int d = (t < k)? 0: 1;
296        path.emplace_back(node);
297        path_d.emplace_back(d);
298        if (d) {
299          node = node->left;
300        } else {
301          k -= t + 1;
302          node = node->right;
303        }
304      }
305      if (node->left && node->right) {
306        node->_propagate();
307        NodePtr lmax = node->left;
308        path.emplace_back(node);
309        path_d.emplace_back(1);
310        lmax->_propagate();
311        while (lmax->right) {
312          path.emplace_back(lmax);
313          path_d.emplace_back(0);
314          lmax = lmax->right;
315          lmax->_propagate();
316        }
317        node->key = lmax->key;
318        node = lmax;
319      }
320      NodePtr cnode = (node->left)? node->left: node->right;
321      if (!path.empty()) {
322        if (path_d.back()) path.back()->left = cnode;
323        else path.back()->right = cnode;
324      } else {
325        this->root = cnode;
326        return res;
327      }
328      while (!path.empty()) {
329        NodePtr new_node = nullptr;
330        node = path.back();
331        node->_update();
332        path.pop_back();
333        path_d.pop_back();
334        double b = node->balance();
335        if (b < ALPHA) {
336          new_node = _balance_left(node);
337        } else if (b > 1.0-ALPHA) {
338          new_node = _balance_right(node);
339        }
340        if (new_node) {
341          if (path.empty()) {
342            this->root = new_node;
343            break;
344          }
345          if (path_d.back()) {
346            path.back()->left = new_node;
347          } else {
348            path.back()->right = new_node;
349          }
350        }
351      }
352      return res;
353    }

355    void reverse(const int l, const int r) {
356      if (l >= r) return;
357      auto [s_, t] = _split_node(root, r);
358      auto [u, s] = _split_node(s_, l);
359      s->rev ^= 1;
360      this->root = _merge_node(_merge_node(u, s), t);

363    vector<T> tovector() {
364      NodePtr node = root;
365      vector<NodePtr> stack;
366      vector<T> a;
367      a.reserve(len());
368      while ((!stack.empty()) || node) {
369        if (node) {
370          node->_propagate();
371          stack.emplace_back(node);
372          node = node->left;
373        } else {
374          node = stack.back();
375          stack.pop_back();
376          a.emplace_back(node->key);
377          node = node->right;
378        }
379      }
380      return a;
381    }

383    void set(int k, T v) {
384      if (k < 0) k += len();
385      assert(0 <= k && k < len());
386      NodePtr node = root;
387      NodePtr root = node;
388      NodePtr pnode = nullptr;
389      int d = 0;
390      vector<NodePtr> path = {node};
391      while (1) {
392        node->_propagate();
393        int t = node->left? node->left->size: 0;
394        if (t == k) {
395          node->key = v;
396          path.emplace_back(node);
397          if (pnode) {
398            if (d) pnode->left = node;
399            else pnode->right = node;
400          }
401          while (!path.empty()) {
402            path.back()->_update();
403            path.pop_back();
404          }
405          return;
406        }
407        pnode = node;
408        d = (t < k)? 0: 1;
409        if (d) {
410          pnode->left = node = node->left;
411        } else {
412          k -= t + 1;
413          pnode->right = node = node->right;
414        }
415        path.emplace_back(node);
416      }
417    }

419    T get(int k) {
420      if (k < 0) k += len();
421      assert(0 <= k && k < len());
422      NodePtr node = root;
423      while (1) {
424        node->_propagate();
425        int t = node->left? node->left->size: 0;
426        if (t == k) {
427          return node->key;
428        }
429        if (t < k) {
430          k -= t + 1;
431          node = node->right;
432        } else {
433          node = node->left;
434        }
435      }
436    }

452    void isok() {
453      auto rec = [&] (auto &&rec, NodePtr node) -> pair<int, int> {
454        int ls = 0, rs = 0;
455        int height = 0;
456        int h;
457        if (node->left) {
458          pair<int, int> res = rec(rec, node->left);
459          ls = res.first;
460          h = res.second;
461          height = max(height, h);
462        }
463        if (node->right) {
464          pair<int, int> res = rec(rec, node->right);
465          rs = res.first;
466          h = res.second;
467          height = max(height, h);
468        }
469        int s = ls + rs + 1;
470        double b = (double)(ls+1) / (s+1);
471        assert(s == node->size);
472        assert(ALPHA <= b && b <= 1-ALPHA);
473        return {s, height+1};
474      };
475      if (root == nullptr) return;
476      auto [_, h] = rec(rec, root);
477      // printf("height=%d\n", h);
478    }

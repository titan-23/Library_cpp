  2#include <vector>
  3#include <cassert>
  4using namespace std;
  5
  6// LinkCutTree
  7namespace titan23 {
  8
  9    class LinkCutTree {
 10      private:
 11        struct Node;
 12        using NodePtr = Node*;
 13        vector<NodePtr> pool;
 14
 15        struct Node {
 16            int index, size, rev;
 17            NodePtr left, right, par;
 18
 19            Node(const int index) :
 20                    index(index), size(1), rev(0),
 21                    left(nullptr), right(nullptr), par(nullptr) {
 22            }
 23
 24            bool is_root() const {
 25                return (!par) || (!(par->left == this || par->right == this));
 26            }
 27        };
 28
 29        void _apply_rev(const NodePtr node) {
 30            if (!node) return;
 31            node->rev ^= 1;
 32        }
 33
 34        void _propagate(const NodePtr node) {
 35            if (!node) return;
 36            if (node->rev) {
 37                swap(node->left, node->right);
 38                _apply_rev(node->left);
 39                _apply_rev(node->right);
 40                node->rev = 0;
 41            }
 42        }
 43
 44        void _update(const NodePtr node) {
 45            if (!node) return;
 46            _propagate(node->left);
 47            _propagate(node->right);
 48            node->size = 1;
 49            if (node->left) {
 50                node->size += node->left->size;
 51            }
 52            if (node->right) {
 53                node->size += node->right->size;
 54            }
 55        }
 56
 57        void _rotate(const NodePtr node) {
 58            const NodePtr pnode = node->par;
 59            const NodePtr gnode = pnode->par;
 60            _propagate(pnode);
 61            _propagate(node);
 62            if (gnode) {
 63                if (gnode->left == pnode) {
 64                    gnode->left = node;
 65                } else if (gnode->right == pnode) {
 66                    gnode->right = node;
 67                }
 68            }
 69            node->par = gnode;
 70            if (pnode->left == node) {
 71                pnode->left = node->right;
 72                if (node->right) node->right->par = pnode;
 73                node->right = pnode;
 74            } else {
 75                pnode->right = node->left;
 76                if (node->left) node->left->par = pnode;
 77                node->left = pnode;
 78            }
 79            pnode->par = node;
 80            _update(pnode);
 81            _update(node);
 82        }
 83
 84        void _splay(const NodePtr node) {
 85            while ((!node->is_root()) && (!node->par->is_root())) {
 86                if ((node->par->par->left == node->par) == (node->par->left == node)) {
 87                    _rotate(node->par);
 88                } else {
 89                    _rotate(node);
 90                }
 91                _rotate(node);
 92            }
 93            if (!node->is_root()) _rotate(node);
 94            _propagate(node);
 95        }
 96
 97        void _link(const NodePtr c, const NodePtr p) {
 98            _expose(c);
 99            _expose(p);
100            c->par = p;
101            p->right = c;
102            _update(p);
103        }
104
105        void _cut(const NodePtr c) {
106            _expose(c);
107            c->left->par = nullptr;
108            c->left = nullptr;
109            _update(c);
110        }
111
112        NodePtr _expose(const NodePtr node) {
113            NodePtr pre = node;
114            while (node->par) {
115                _splay(node);
116                node->right = nullptr;
117                _update(node);
118                if (!node->par) break;
119                pre = node->par;
120                _splay(node->par);
121                node->par->right = node;
122                _update(node->par);
123            }
124            node->right = nullptr;
125            _update(node);
126            return pre;
127        }
128
129        NodePtr _root(NodePtr v) {
130            _expose(v);
131            _propagate(v);
132            while (v->left) {
133                v = v->left;
134                _propagate(v);
135            }
136            _splay(v);
137            return v;
138        }
139
140        void _evert(NodePtr v) {
141            _expose(v);
142            _apply_rev(v);
143            _propagate(v);
144        }
145
146      public:
147        LinkCutTree() {}
148
149        LinkCutTree(int n) {
150            pool.resize(n);
151            for (int i = 0; i < n; ++i) {
152                pool[i] = new Node(i);
153            }
154        }
155
156        int expose(int v) {
157            return _expose(pool[v])->index;
158        }
159
160        int lca(int u, int v, int root=-1) {
161            if (root != -1) evert(root);
162            expose(u);
163            return expose(v);
164        }
165
166        void link(int c, int p) {
167            _link(pool[c], pool[p]);
168        }
169
170        void cut(int c) {
171            _cut(pool[c]);
172        }
173
174        int root(int v) {
175            return _root(pool[v])->index;
176        }
177
178        bool same(int u, int v) {
179            return root(u) == root(v);
180        }
181
182        void evert(int v) {
183            _evert(pool[v]);
184        }
185
186        bool merge(int u, int v) {
187            if (same(u, v)) return false;
188            evert(u);
189            link(u, v);
190            return true;
191        }
192
193        void split(int u, int v) {
194            evert(u);
195            cut(v);
196        }
197
198        int path_kth_elm(int s, int t, int k) {
199            evert(s);
200            expose(t);
201            NodePtr node = pool[t];
202            if (node->size <= k) return -1;
203            while (true) {
204                _propagate(node);
205                t = node->left? node->left->size: 0;
206                if (t == k) {
207                    _splay(node);
208                    return node->index;
209                }
210                if (t > k) {
211                    node = node->left;
212                } else {
213                    node = node->right;
214                    k -= t + 1;
215                }
216            }
217        }
218    };
219}  // namespace titan23

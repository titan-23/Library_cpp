  8    template <class T,
  9              T (*_op)(T, T),
 10              T (*_e)()>
 11    struct SegmentTree {
 12      private:
 13        int n, _size, _log;
 14        vector<T> _data;
 15
 16      public:
 17        SegmentTree() {}
 18
 19        SegmentTree(const int n) {
 20            _build(n);
 21        }
 22
 23        SegmentTree(const vector<T> &a) {
 24            int n = (int)a.size();
 25            _build(n);
 26            for (int i = 0; i < n; ++i) {
 27                _data[i+_size] = a[i];
 28            }
 29            for (int i = _size-1; i > 0; --i) {
 30                _data[i] = _op(_data[i<<1], _data[i<<1|1]);
 31            }
 32        }
 33
 34        void _build(const int n) {
 35            this->n = n;
 36            this->_log = 32 - __builtin_clz(n);
 37            this->_size = 1 << _log;
 38            this->_data.resize(_size << 1, _e());
 39        }
 40
 41        T get(int const k) const {
 42            return _data[(k<0? (k+n+_size): (k+_size))];
 43        }
 44
 45        void set(int k, const T v) {
 46            if (k < 0) k += n;
 47            k += _size;
 48            _data[k] = v;
 49            for (int i = 0; i < _log; ++i) {
 50                k >>= 1;
 51                _data[k] = _op(_data[k<<1], _data[k<<1|1]);
 52            }
 53        }
 54
 55        T prod(int l, int r) const {
 56            l += _size;
 57            r += _size;
 58            T lres = _e(), rres = _e();
 59            while (l < r) {
 60                if (l & 1) lres = _op(lres, _data[l++]);
 61                if (r & 1) rres = _op(_data[r^1], rres);
 62                l >>= 1;
 63                r >>= 1;
 64            }
 65            return _op(lres, rres);
 66        }
 67
 68        T all_prod() const {
 69            return _data[1];
 70        }
 71
 72        template<typename F>  // F: function<bool (T)> f
 73        int max_right(int l, F &&f) const {
 74            assert(0 <= l && l <= _size);
 75            // assert(f(_e()));
 76            if (l == n) return n;
 77            l += _size;
 78            T s = _e();
 79            while (1) {
 80                while ((l & 1) == 0) {
 81                l >>= 1;
 82                }
 83                if (!f(_op(s, _data[l]))) {
 84                while (l < _size) {
 85                    l <<= 1;
 86                    if (f(_op(s, _data[l]))) {
 87                    s = _op(s, _data[l]);
 88                    l |= 1;
 89                    }
 90                }
 91                return l - _size;
 92                }
 93                s = _op(s, _data[l]);
 94                ++l;
 95                if ((l & (-l)) == l) break;
 96            }
 97            return n;
 98        }
 99
100        template<typename F>  // F: function<bool (T)> f
101        int min_left(int r, F &&f) const {
102            assert(0 <= r && r <= n);
103            // assert(f(_e()));
104            if (r == 0) return 0;
105            r += _size;
106            T s = _e();
107            while (r > 0) {
108                --r;
109                while (r > 1 && (r & 1)) {
110                r >>= 1;
111                }
112                if (!f(_op(_data[r], s))) {
113                while (r < _size) {
114                    r = (r << 1) | 1;
115                    if (f(_op(_data[r], s))) {
116                    s = _op(_data[r], s);
117                    r ^= 1;
118                    }
119                }
120                return r + 1 - _size;
121                }
122                s = _op(_data[r], s);
123                if ((r & (-r)) == r) break;
124            }
125            return 0;
126        }
127
128        vector<T> tovector() const {
129            vector<T> res(n);
130            for (int i = 0; i < n; ++i) {
131                res[i] = get(i);
132            }
133            return res;
134        }
135
136        void print() const {
137            cout << '[';
138            for (int i = 0; i < n-1; ++i) {
139                cout << get(i) << ", ";
140            }
141            if (n > 0) cout << get(n-1);
142            cout << ']' << endl;
143        }
144    };
145}  // namespace titan23

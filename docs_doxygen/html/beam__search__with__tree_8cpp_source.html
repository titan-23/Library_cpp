399  pair<int, bool> get_pos(const u64 &key) const {
400  int h = hash(key);
401  while (true) {
402  if (!(exist[h>>6]>>(h&63)&1)) return {h, false};
403  if (keys[h] == key) return {h, true};
404  h = (h + 1) & msk;
405  }
406  }

408  int bit_length(const int x) const {
409  if (x == 0) return 0;
410  return 32 - __builtin_clz(x);
411  }

413  void rebuild() {
414  vector<u64> old_exist = exist;
415  vector<u64> old_keys = keys;
416  exist.resize(HashSet::M*old_exist.size()+1);
417  fill(exist.begin(), exist.end(), 0);
418  keys.resize(HashSet::M*old_keys.size());
419  size = 0;
420  msk = (1<<bit_length(keys.size()-1))-1;
421  random_device rd;
422  mt19937 gen(rd());
423  uniform_int_distribution<int> dis(0, msk);
424  xor_ = dis(gen);
425  for (int i = 0; i < (int)old_keys.size(); ++i) {
426  if (old_exist[i>>6]>>(i&63)&1) {
427  insert(old_keys[i]);
428  }
429  }
430  }

435  HashSet(const int n) {

442  random_device rd;

444  uniform_int_distribution<int> dis(0, msk);
